      // note: prepend this script to <body> because:
      // - the Angular CLI appends dependencies to the <body> that throw error in IE11
      // - this script needs to reference the <body>
      (function validateBrowser() {
        // if the browser is not IE11
        if (!window['MSInputMethodContext'] || !document['documentMode']) {
          // then no-op
          return;
        }

        // if the browser is IE11,
        // then generate the IE11 error page
        const anchors = [
          ['https://www.google.com/chrome', 'Google Chrome'],
          ['https://www.mozilla.org/firefox/new', 'Mozilla Firefox'],
          ['https://www.microsoft.com/edge', 'Microsoft Edge'],
        ];
        const anchorsHtml = anchors
          .reduce(function (acc, browser, idx) {
            const anchorParts = ['<a class="ie-11-message__cta" href="', browser.join('" target="_blank">'), '</a>'];
            // if there are multiple anchors, and
            // if the current anchor is the second last
            if (anchors.length > 1 && idx === anchors.length - 2) {
              // then insert 'or'
              anchorParts.push('<span> or </span>');
            }
            return acc.concat(anchorParts.join(''));
          }, [])
          .join(' ');

        const htmlBodyElement = document.querySelector('body');
        htmlBodyElement.innerHTML = [
          '<section class="ie-11-message"> <section class="ie-11-message__header"> Browser not supported </section> <section class="ie-11-message__body"> <span> We suggest using the latest version of </span>',
          anchorsHtml,
          '</section> </section>',
        ].join(' ');
        htmlBodyElement.classList.add('ie-11-message-container');
      })();
    </script>

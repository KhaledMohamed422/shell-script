<script>
    function isIE() {
        ua = navigator.userAgent;
        var is_ie = ua.indexOf("MSIE ") > -1 || ua.indexOf("Trident/") > -1;
        return is_ie;
    }

    function setTheStyle() {
        if (isIE()) {
            document.getElementById('Link_1').style.color = '#2cdd9d';
            document.getElementById('Link_1').style.background = 'none';

            document.getElementById('Link_2').style.color = '#2cdd9d';
            document.getElementById('Link_2').style.background = 'none';
        }
    }

    function IframeLoaded() {
        let originalHeight, originalWidth, originalZindex;
        originalHeight = $("#zendeskBot").css("height");
        originalWidth = $("#zendeskBot").css("width");
        originalZindex = $("#zendeskBot").css("zIndex");
        let shouldCalllauncherInterval = true;
        let shouldCallInsideInterval = true;
        let shouldCallWidgetInterval = true;
        setIntervalAsync(outerframeCallback, 200, 'launcherInterval');
        function outerframeCallback() {
            if ($("#zendeskBot").contents().find("iframe").length > 1) {
                shouldCalllauncherInterval = false;
                outerIframeDone($("#zendeskBot").contents().find("iframe")[1]);
            }
        };

        function outerIframeDone(launcherIframe) {
            setIntervalAsync(launcherCallback, 200, 'insideInterval');
            function launcherCallback() {
                if ($(launcherIframe).contents().find("#Embed").length > 0) {
                    shouldCallInsideInterval = false;
                }
                bindBotHeightToEmbed($(launcherIframe).contents().find("#Embed"));
                $("#zendeskBot").css("display", "block");
            }
        };

        function bindBotHeightToEmbed(iframeLauncher) {
            iframeLauncher.on("click", function () {
                shouldCallWidgetInterval = true;
                $("#zendeskBot").css("height", "100%").css("width", "100%").css("zIndex", "200");
                setIntervalAsync(webwidgetCallback, 200, 'widgetInterval');
                function webwidgetCallback() {
                    let webwidget = $("#zendeskBot").contents().find("iframe")[2];
                    if (webwidget) {
                        if ($(webwidget).contents().find('[aria-label="Minimise widget"],[aria-label="Minimize widget"]').length > 0) {
                            shouldCallWidgetInterval = false;
                            bindResetBotHeightToButton(webwidget);
                        }
                    }
                };

            });
        };

        function bindResetBotHeightToButton(webwidget) {
            $(webwidget).contents().on("pointerup",'[aria-label="Minimise widget"],[aria-label="Minimize widget"]', function (bindEvent) {
                shouldCallWidgetInterval = false;
                shouldCalllauncherInterval = false;
                shouldCallInsideInterval = false;                
                $("#zendeskBot").css("height", originalHeight).css("width", originalWidth).css("zIndex", originalZindex);                
                $(bindEvent.currentTarget).trigger("click");                
            });
        };
        function setIntervalAsync(fn, ms, interval) {
            fn();
            const myPromise = new Promise((resolve, reject) => {
                if (interval === "launcherInterval" && shouldCalllauncherInterval) {
                    setTimeout(() => setIntervalAsync(fn, ms, interval), ms);
                    resolve(interval);
                }
                else if (interval === "insideInterval" && shouldCallInsideInterval) {
                    setTimeout(() => setIntervalAsync(fn, ms, interval), ms);
                    resolve(interval);
                }
                else if (interval === "widgetInterval" && shouldCallWidgetInterval) {
                    setTimeout(() => setIntervalAsync(fn, ms, interval), ms);
                    resolve(interval);
                }
            });
        };
    }

<script>
const URL = 'https://id.indrive.com/api/contests';
const CONTEST = 'webflow_test_contests';
const DATA_END = '2023-05-31';
const TRIPS_TO_RAFFLE = 45;
const MESSAGES = {
    error: {
        participantNotFound: 'Пользователь не найден. Проверьте введенные данные',
        serverError: 'Возникли проблемы на сервере, Повторите запрос позднее',
        fieldCityIsEmpty: 'Укажите город',
        fieldIdIsEmpty: 'Поле ID не может быть пустым',
    },
    title: {
        conditionIsMet: 'Вы выполнили все поездки!',
        contestIsOver: 'Конкурс завершён',
    }
};

const singUp = {
    element: document.querySelector('.form-sign-up'),
    form: document.querySelector('.form-sign-up form'),
    inputId: document.querySelector('#id'),
    selectCity: document.querySelector('#field'),
    submitBtn: document.querySelector('.form-sign-up input[type="Submit"]'),
};

const logIn = {
    element: document.querySelector('.form-login'),
    form: document.querySelector('.form-login form'),
    inputId: document.querySelector('#id-2'),
    submitBtn: document.querySelector('.form-login input[type="Submit"]'),
};

const result = {
    element: document.querySelector('.block-result'),
    trips: document.querySelector('#rides'),
    progress: document.querySelector('.progress-bar__scale'),
    title: document.querySelector('.block-result-title h1'),
    image: document.querySelector('.block-result-image'),
};

singUp.form.addEventListener('submit', (e) => {
    e.preventDefault();
    e.stopPropagation();

    singUp.submitBtn.disabled = true;
    removeError(e.target);

    const formData = {
        id: encodeURIComponent(singUp.inputId.value),
        city: encodeURIComponent(singUp.selectCity.value),
        referer: window.location.toString(),
    };
    const errors = validation.singUp(formData);

    if (errors.length > 0) {
        setError(singUp.form, errors);
        singUp.submitBtn.disabled = false;
        return;
    }

    axios.post(`${URL}/${CONTEST}`, formData)
        .then((response) => {
            hide(singUp.element);
            showProgress(response.data);
        })
        .catch((error) => {
            const message = axiosErrorHandler(error);
            setError(singUp.form, [message]);
            singUp.submitBtn.disabled = false;
        });
});


logIn.form.addEventListener('submit', async (e) => {
    e.preventDefault();
    e.stopPropagation();

    logIn.submitBtn.disabled = true;
    removeError(e.target);

    const id = logIn.inputId.value;
    const errors = validation.logIn(id);

    if (errors.length > 0) {
        setError(logIn.form, errors);
        logIn.submitBtn.disabled = false;
        return;
    }

    axios.get(`${URL}/${CONTEST}/${id}`)
        .then((response) => {
            hide(logIn.element);
            showProgress(response.data);
        })
        .catch((error) => {
            const message = axiosErrorHandler(error);
            setError(logIn.form,[message]);
            logIn.submitBtn.disabled = false;
        });
});

document.addEventListener('DOMContentLoaded', setID);

function showProgress (data){
    const numberTrips = data.number_trips ? Number(data.number_trips) : 0;
    const percent = Math.round(numberTrips * 100 / TRIPS_TO_RAFFLE) >= 100
        ? 100 : Math.round(numberTrips  * 100 / TRIPS_TO_RAFFLE);

    result.trips.textContent = `${numberTrips} / ${TRIPS_TO_RAFFLE}`;
    result.element.classList.add('block-result--visible');

    setTimeout(() => {
        result.progress.style.width = percent + '%';
    }, 100);

    if (new Date() > new Date(DATA_END)) {
        if (numberTrips === 0) hide(result.progress);
        if (numberTrips < TRIPS_TO_RAFFLE) {
            result.title.textContent = MESSAGES.title.contestIsOver;
            result.progress.style.background = '#BDBDBD';
            result.element.querySelector("#message_7").classList.add('block-result-message--visible');
            return;
        }
        result.title.textContent = MESSAGES.title.conditionIsMet;
        result.image.classList.add('block-result-image--visible');
        result.element.querySelector("#message_6").classList.add('block-result-message--visible');
        return;
    }

    switch(true) {
        case (numberTrips < 1):
            result.element.querySelector("#message_1").classList.add('block-result-message--visible');
            hide(result.progress);
            break;
        case (numberTrips < 11):
            result.element.querySelector("#message_2").classList.add('block-result-message--visible');
            break;
        case (numberTrips < 26):
            result.element.querySelector("#message_3").classList.add('block-result-message--visible');
            break;
        case (numberTrips < 36):
            result.element.querySelector("#message_4").classList.add('block-result-message--visible');
            break;
        case (numberTrips < 45):
            result.element.querySelector("#message_5").classList.add('block-result-message--visible');
            break;
        default:
            result.title.textContent = MESSAGES.title.conditionIsMet;
            result.image.classList.add('block-result-image--visible');
            result.element.querySelector("#message_6").classList.add('block-result-message--visible');
    }
}

function setID() {
    const allParams = new URLSearchParams(window.location.search);
    const uid = allParams.get('uid');
    singUp.inputId.value = uid;
    logIn.inputId.value = uid;
}

const validation = {
    singUp: (data) => {
        const errors = [];
        if (data.id === '') errors.push(MESSAGES.error.fieldIdIsEmpty);
        if (data.city === '') errors.push(MESSAGES.error.fieldCityIsEmpty);
        return errors;
    },
    logIn: (id) => {
        const errors = [];
        if (id === '') errors.push(MESSAGES.error.fieldIdIsEmpty);
        return errors;
    }
};

function setError (form, errors){
    const feedback = document.createElement('div');
    feedback.classList.add('w-form-fail', 'w-form-fail--visible');

    errors.forEach((message) => {
        const el = document.createElement('div');
        el.textContent = message;
        feedback.prepend(el);
    });

    form.after(feedback);
}

function removeError (){
    document.querySelectorAll('.w-form-fail--visible')
        .forEach((feedback) => feedback.remove());
}

function axiosErrorHandler (error){
    if (error.response) {
        if (error.response.status === 404) return MESSAGES.error.participantNotFound;
    }
    return MESSAGES.error.serverError;
}

function hide (element){
    element.style.display = 'none';
}

function refreshPage(){
    location.reload();
}

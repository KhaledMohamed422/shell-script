            <script>
    var host = window.location.host;
    var parts = host.split(".");
    var hostPartsLen = parts.length;
    var mainHost = ""
    for(var i=0; i<hostPartsLen;i++){
        if (i == 0){
            mainHost = parts[i]
        }
        else if(i== 1){
            continue;
        }else {
            mainHost = mainHost + "." + parts[i];
        }
    }

    var processATag = function(node, index){
        var href = node.href;
        if (href.match(mainHost)) {
            node.href = href.replace(mainHost, host);
        }
    }

    var processFormTag = function(node, index){
        var action = node.action;
        if (action.match(mainHost)) {
            node.action = action.replace(mainHost, host);
        }
    }
    KISSY.use('node', function (S, Node) {
        var hrefNodes = Node.all("[href]");
        S.each(hrefNodes, processATag);

        var formNodes = Node.all("form");
        S.each(formNodes, processFormTag);

        Node.all('.J_TModule').on('click', function (e) {
            var urlReplaced = this.getAttribute("data-worldsite-url-replaced");

            if(!urlReplaced){
                var id = this.id;
                var prefix = "#" + id;

                var aTagNodes = S.all(prefix + " a");
                S.each(aTagNodes, processATag);

                var formTagNodes = S.all(prefix + " form");
                S.each(formTagNodes, processFormTag);

                this.setAttribute("data-worldsite-url-replaced",true);
            }


        });
    });
</script>

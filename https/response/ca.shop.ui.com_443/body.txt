Vue.component('cart-modal', {
  template: '#cart-modal',

  props: {
    open: Boolean,
    cartAccessories: Array,
  },

  delimiters: ['${', '}'],

  data() {
    return {
      loading: false,
      cart: CartJS.cart,
      tmpScrollTop: 0,
      mobileModes: {
        ITEMS: 0,
        SUGGESTED: 1,
      },
      mobileMode: 0,
      expandedProductIds: [],
      wasQuantityReduced: false,
    };
  },

  computed: {
    /**
     * @returns {Object}
     */
    customer() {
      return this.$root.appData.customer;
    },

    /**
     * Cart items shortcut, always returns an array
     *
     * @returns {Array}
     */
    cartItems() {
      return this.cart.items || [];
    },

    /**
     * Check if there are items in the cart
     *
     * @returns {boolean}
     */
    isCartEmpty() {
      return this.cartItems.length === 0;
    },

    /**
     * Get suggested products
     *
     * @returns {Array}
     */
    suggestedProducts() {
      // Return no products if the cart is empty - do not suggest anything
      if (this.isCartEmpty) {
        return [];
      }

      // Clone the accessories
      let accessories = this.cartAccessories.slice(0);

      // Sort based on the priority
      accessories = accessories.sort(
        (a, b) => parseFloat(b.data.priority || 1) - parseFloat(a.data.priority || 1),
      );

      return accessories;
    },

    /**
     * Get suggested product variants that are available
     *
     * @returns {Array}
     */
    suggestedVariants() {
      let variants = [];
      const maxVariantCount = 15;

      const checkView = viewData => {
        // Legacy boolean value
        if (!Array.isArray(viewData)) {
          return false;
        }

        // Ignored SKU not in cart
        if (viewData.some(sku => this.cartItems.some(item => item.sku === `!${sku}`))) {
          return false;
        }

        // Wildcard
        if (viewData.includes('*')) {
          return true;
        }

        // SKU in cart
        return viewData.some(sku => this.cartItems.some(item => item.sku === sku));
      };

      for (const product of this.suggestedProducts) {
        for (const variant of product.variants) {
          const isInCart = this.cartItems.some(
            item => item.sku === variant.sku,
          );

          // Add product object reference to the variant
          variant.product = product;

          // Add the variant to the list if not already in the cart
          if (!isInCart) {
            variants.push(variant);
          }
        }
      }

      // Hide variants that are not allowed for cart items
      variants = variants.filter(variant => {
        const data = variant.product.data || {};
        const dataFor = data.for || {};

        const checkFor = (dataKey, itemKey) => {
          if (dataFor[dataKey] && dataFor[dataKey].length) {
            let found = false;

            dataFor[dataKey].forEach(value => {
              if (this.cartItems.some(item => item[itemKey] === value)) {
                found = true;
                return;
              }
            });

            // Valid if targetted product is found in the cart
            return found;
          }

          // Valid by default
          return true;
        };

        // Check the country validity
        if (
          Array.isArray(data.countries) &&
          data.countries.length &&
          !data.countries.includes('US')
        ) {
          return false;
        }

        // Check for product match
        if (data.view && data.view.checkout && !checkView(data.view.checkout)) {
          return false;
        }

        // Lastly, check for targetted products
        return (
          checkFor('product-vendors', 'vendor') &&
          checkFor('product-types', 'product_type')
        );
      });

      // Limit variant count
      variants.splice(maxVariantCount);

      return variants;
    },

    /**
     * Check for suggested product count
     *
     * @returns {boolean}
     */
    hasSuggestedProducts() {
      return this.suggestedProducts.length > 0;
    },

    /**
     * Check for suggested product variant count

    /**
     * Validate cart quantities
     * Attribute are set in vue-product-cart-relations
     */
    async checkCartProductQuantities() {
      const { items } = this.cart;
      const limitsAttribute = this.$root.getCartAttribute('quantity-limits');
      const kitsAttribute = this.$root.getCartAttribute('quantity-kits');
      const hddsAttribute = this.$root.getCartAttribute('quantity-hdds');

      const itemQuantities = {};
      const debugQuantityChanges = false;
      const setItemQuantity = (itemId, quantity, reason = '') => {
        if (debugQuantityChanges) {
          console.log(`${itemId} was set as ${quantity}`, reason ? `: ${reason}` : '');
        }
        itemQuantities[itemId] = quantity;
      };

      // Manage HDDs
      const cartHddIdMaxQties = {};
      const allCartHddIds = [];

      for (const [dependantProductId, hdds] of Object.entries(hddsAttribute)) {
        const dependantCartItem = items.find(
          ({ product_id: productId }) =>
            `${productId}` === `${dependantProductId}`,
        );
        const hddSkus = Object.values(hdds).map(({ sku }) => sku);

        items
          .filter(({ sku }) => hddSkus.includes(sku))
          .forEach(({ id, sku }) => {
            // Store the HDD id
            if (!allCartHddIds.includes(id)) {
              allCartHddIds.push(id);
            }

            // Set the max allowed quantity if possible
            if (dependantCartItem) {
              const hdd = Object.values(hdds).find(hdd => hdd.sku === sku);
              const prevQty = parseInt(cartHddIdMaxQties[id]) || 0;
              const dependantQty = parseInt(dependantCartItem.quantity);
              const ratio = (hdd && parseInt(hdd.ratio)) || 1;

              cartHddIdMaxQties[id] = prevQty + dependantQty * ratio;
            }
          });
      }

      for (const cartHddId of allCartHddIds) {
        const cartHdd = items.find(({ id }) => `${id}` === `${cartHddId}`);
        const cartHddMaxQty = parseInt(cartHddIdMaxQties[cartHddId]) || 0;

        if (cartHdd && cartHdd.quantity > cartHddMaxQty) {
          // Limit the allowed max quantity
          setItemQuantity(cartHddId, cartHddMaxQty, 'HDD max limit');
        }
      }

      // Manage kits
      for (const [parentProductId, kitItems] of Object.entries(kitsAttribute)) {
        const parentCartItem = items.find(
          ({ product_id: productId }) =>
            `${productId}` === `${parentProductId}`,
        );
        const parentProductLimits = limitsAttribute[parentProductId];

        if (parentCartItem) {
          for (const kitItem of Object.values(kitItems)) {
            // Add the kit item to the cart if not there already
            if (!items.some(({ id }) => `${id}` === `${kitItem.itemId}`)) {
              const quantity = parseInt(kitItem.quantity) || 1;

              setItemQuantity(kitItem.itemId, quantity * parentCartItem.quantity, 'Kit item');
            }
          }

          // Remove the parent item from cart
          setItemQuantity(parentCartItem.id, 0, 'Kit parent item');
        } else {
          // Check and manage the cart kit items
          const kitItemIds = Object.values(kitItems).map(
            ({ itemId }) => `${itemId}`,
          );
          const kitCartItems = items.filter(({ id }) =>
            kitItemIds.includes(`${id}`),
          );

          if (kitCartItems.length < kitItemIds.length) {
            // Remove dangling kit items
            for (const kitItem of Object.values(kitItems)) {
              if (
                kitCartItems.some(({ id }) => `${id}` === `${kitItem.itemId}`)
              ) {
                setItemQuantity(kitItem.itemId, 0, 'Dangling kit item');
              }
            }
          } else {
            // Manage kit item quantity linking
            const kitCartQties = kitCartItems.map(({ quantity }) =>
              parseInt(quantity),
            );
            const kitCartQtiesUnique = [...new Set(kitCartQties)];

            if (kitCartQtiesUnique.length > 1) {
              const quantity = this.wasQuantityReduced
                ? Math.min(...kitCartQtiesUnique)
                : Math.max(...kitCartQtiesUnique);

              for (const kitItemId of kitItemIds) {
                setItemQuantity(kitItemId, quantity, 'Equalize kit item quantities');
              }
            }
          }
        }

        // Adjust the quantity limits based on the parent
        if (parentProductLimits) {
          const { minimum, maximum } = parentProductLimits;

          for (const { itemId } of Object.values(kitItems)) {
            const quantity = itemQuantities[itemId];

            if (maximum && quantity > maximum) {
              setItemQuantity(itemId, maximum, 'Max allowed parent qty');
            } else if (minimum && quantity < minimum) {
              setItemQuantity(itemId, minimum, 'Min allowed parent qty');
            }
          }
        }
      }

      // Manage limits - has to be the last one
      for (const [limitProductId, limits] of Object.entries(limitsAttribute)) {
        const minimum = parseInt(limits.minimum) || this.$root.defaultQtyMinimum;
        const maximum = parseInt(limits.maximum) || this.$root.defaultQtyMaximum;
        const multiple = parseInt(limits.multiple) || null;

        for (const item of items) {
          const { id, product_id: productId } = item;
          let { quantity } = item

          if (`${productId}` === `${limitProductId}`) {
            // Enforce step - has to be the first one
            if (multiple !== null && multiple >= 0 && quantity % multiple !== 0) {
              const diff = quantity % multiple;
              const additive = this.wasQuantityReduced ? -diff : multiple - diff;

              quantity += additive;
              setItemQuantity(id, quantity, 'Multi step quantity');
            }

            // Enforce max quantity
            if (maximum !== null && maximum >= 0 && quantity > maximum) {
              setItemQuantity(id, maximum, 'Max allowed qty');
            }

            // Enforce min quantity
            if (minimum !== null && minimum >= 0 && quantity < minimum) {
              // Allow setting it to 0 always
              setItemQuantity(id, quantity > 0 ? minimum : 0, 'Min allowed quantity');
            }
          }
        }
      }

      // Per-customer limits
      const customerProductLimits = await this.$root.getCustomerProductLimits();

      if (customerProductLimits) {
        const notLoggedIn = !(this.customer.email && this.customer.email.length);
        const boughtItemQties = this.$root.getBoughtItemQtiesGroupedBySku();

        for (const item of items) {
          const { id, quantity } = item;
          const variantMaxQtyLimit = customerProductLimits.find(
            limit => limit.sku.toLowerCase() === item.sku.toLowerCase()
          );

          if (!variantMaxQtyLimit) {
            continue;
          }

          if (notLoggedIn) {
            // Set the item quantity to 0 due to requiring a login
            setItemQuantity(id, 0, 'Customer product limits requires logging in');
            continue;
          }

          const boughtQty = boughtItemQties[item.sku] || 0;
          const customerAllowedQty = variantMaxQtyLimit.max - boughtQty;

          if (quantity > customerAllowedQty) {
            // Update the item quantity to the allowed limit
            const finalItemQty = customerAllowedQty >= 0 ? customerAllowedQty : 0;

            setItemQuantity(id, finalItemQty, 'Customer product limits max allowed');
          }
        }
      }

      // Final maximum quantity limit
      for (const item of items) {
        const { id, quantity } = item;
        const maxQuantity = this.$root.defaultQtyMaximum;

        if (quantity > maxQuantity) {
          setItemQuantity(id, maxQuantity, 'Final max quantity');
        }
      }

      // Update quantities
      if (Object.values(itemQuantities).length) {
        // If any product is sold out or fails, all updates are cancelled so it's individual updates
        for (const [itemId, quantity] of Object.entries(itemQuantities)) {
          if (/^[0-9]+$/.test(quantity)) {
            CartJS.updateItemQuantitiesById({ [itemId]: quantity });
          } else {
            console.error(`Could not update ${itemId} to ${quantity}`);
          }
        }
      }
    },

    /**
     * Redirect to checkout
     */
    navigateToCheckout() {
      if (!this.isCartEmpty) {
        this.loading = true;
        window.location = '/checkout';
      }
    },

    /**
     * Scroll the items list to the top
     */
    scrollItemsToTop() {
      const { itemsList } = this.$refs;

      if (itemsList) {
        jQuery(itemsList).animate({ scrollTop: 0 }, 300);
      }
    },

    /**
     * Update target element text to match cart total item count
     */
    updateCartItemCountEls() {
      const { item_count: count } = this.cart;
      const countEls = document.querySelectorAll('.cart-item-count');

      if (countEls) {
        countEls.forEach(el => {
          el.innerText = count;
        });
      }
    },

    removeSoldOutItems() {
      const [ lastAddedItem ] = this.cart.items
      const isSoldOutCheck = product => product.tags.split(', ').some(tag => tag === 'sold-out')
      const removeLastAddedItem = () => {
        CartJS.removeItemById(lastAddedItem.id)
      }
      if(lastAddedItem && lastAddedItem.handle) {
        fetch(`/products/${lastAddedItem.handle}.json`)
          .then(data => data.json())
          .then(({ product }) => isSoldOutCheck(product))
          .then(isSoldOut => isSoldOut && removeLastAddedItem())
      }
    },

    onCartOpen() {
      const { id, items } = this.cart

      // Track
      Segment.track('Cart Viewed', {
        cart_id: id,
        products: items.map((item, i) => ({
          product_id: item.product_id,
          sku: item.sku,
          category: item.product_type,
          name: item.product_title,
          brand: item.vendor,
          variant: item.variant_id,
          price: Segment.centsToMoney(item.price),
          quantity: item.quantity,
          coupon: '',
          position: i,
          url: `https://${window.location.host}${item.url}`,
          image_url: item.image,
        })),
      })

      // Disable body scrolling
      this.tmpScrollTop = jQuery(window).scrollTop()
      jQuery('body').css('overflow', 'hidden')

      // Reset the mobile mode to items
      this.setMobileMode(this.mobileModes.ITEMS)

      // Reset scroll
      this.scrollItemsToTop()
    },

    onCartClose() {
      // Enable body scrolling
      jQuery('body').css('overflow', 'auto')
      jQuery(window).scrollTop(this.tmpScrollTop)
    },

    onItemAdd(item, itemIndex) {
      // Track
      Segment.track('Product Added', {
        cart_id: '',
        product_id: item.product_id,
        sku: item.sku,
        category: item.product_type,
        name: item.product_title,
        brand: item.vendor,
        variant: item.variant_id,
        price: Segment.centsToMoney(item.price),
        quantity: item.quantity,
        coupon: '',
        position: itemIndex,
        url: `https://${window.location.host}${item.url}`,
        image_url: item.image,
      })
    },

    onItemRemove(item, itemIndex) {
      // Track
      Segment.track('Product Removed', {
        cart_id: '',
        product_id: item.product_id,
        sku: item.sku,
        category: item.product_type,
        name: item.product_title,
        brand: item.vendor,
        variant: item.variant_id,
        price: Segment.centsToMoney(item.price),
        quantity: item.quantity,
        coupon: '',
        position: itemIndex,
        url: `https://${window.location.host}${item.url}`,
        image_url: item.image,
      })
    },
  },

  watch: {
    /**
     * @param {Array} items
     * @param {Array} prevItems
     */
    cartItems(items, prevItems) {
      // Update counts
      this.updateCartItemCountEls()

      // Update quantity change check
      const totalQty = items
        .map(({ quantity }) => parseInt(quantity))
        .reduce((a, b) => a + b, 0)
      const prevTotalQty = prevItems
        .map(({ quantity }) => parseInt(quantity))
        .reduce((a, b) => a + b, 0)

      this.wasQuantityReduced = totalQty < prevTotalQty

      // On items change
      const addedItems = items.filter(
        (item) => !prevItems.map((x) => x.id).includes(item.id),
      )
      const removedItems = prevItems.filter(
        (item) => !items.map((x) => x.id).includes(item.id),
      )

      for (const item of addedItems) {
        const itemIndex = items.findIndex(({ id }) => id === item.id)

        this.onItemAdd(item, itemIndex)
      }

      for (const item of removedItems) {
        const itemIndex = prevItems.findIndex(({ id }) => id === item.id)

        this.onItemRemove(item, itemIndex)
      }

      if (addedItems.length) {
        // Scroll items to top if added
        this.scrollItemsToTop()

        // Remove item if it has "sold-out" tag
        this.removeSoldOutItems()
      }

      // Check cart product links and limits
      this.checkCartProductQuantities()
    },

    /**
     * @param {boolean} state
     */
    open(state) {
      if (state) {
        this.onCartOpen()
      } else {
        this.onCartClose()
      }
    },

    /**
     * @param {Array} variants
     */
    suggestedVariants(variants) {
      if (!variants.length) {
        // Switch to items mode when no suggested items remain
        this.setMobileMode(this.mobileModes.ITEMS);
      }
    },
  },

  mounted() {
    // Body ready actions
    jQuery(() => {
      this.updateCartItemCountEls();
    });

    // Cart events
    jQuery(document).on('cart.requestStarted', this.onBeforeCartUpdate);
    jQuery(document).on('cart.requestComplete', this.onCartUpdate);
  },

  beforeDestroy() {
    jQuery(document).off('cart.requestStarted', this.onBeforeCartUpdate);
    jQuery(document).off('cart.requestComplete', this.onCartUpdate);
  },
});
</script>

Vue.component('cart-banner', {
  template: '#cart-banner',

  props: {
    show: Boolean,
  },

  delimiters: ['${', '}'],

  data() {
    return {
      loading: false,
      cart: CartJS.cart,
      delayedShow: this.show,
    }
  },

  computed: {
    /**
     * Disable transition animation on small devices
     *
     * @returns {number}
     */
    transitionDuration() {
      const { isMobileResolution, isTabletResolution } = this.$root

      return isMobileResolution || isTabletResolution ? 0 : 300
    },

    /**
     * Check if the banner can be shown
     *
     * @returns {boolean}
     */
    canShow() {
      return this.delayedShow && !this.isCartEmpty
    },

    /**
     * Cart items shortcut, always returns an array
     *
     * @returns {Array}
     */
    cartItems() {
      return this.cart.items || []
    },

    /**
     * Check if there are items in the cart
     *
     * @returns {boolean}
     */
    isCartEmpty() {
      return this.cartItems.length === 0
    },
  },

  methods: {
    /**
     * Send a custom close event for the parent to handle
     */
    closeBanner() {
      this.$emit('close')
    },

    /**
     * Enable loading when the cart is about to update
     */
    onBeforeCartUpdate() {
      this.loading = true
    },

    /**
     * Update component data on cart update
     *
     * @param {Object} event
     * @param {Object} cart
     */
    onCartUpdate(event, cart) {
      this.cart = cart
      this.loading = false
    },

    /**
     * Additional page container padding for the banner to not overlap content
     */
    setPageContainerPadding() {
      const containerEl = document.getElementById('PageContainer')

      if (containerEl) {
        containerEl.style.paddingBottom = this.canShow
          ? `${this.$el.offsetHeight}px`
          : 0
      }
    },

    /**
     * Scroll the items list to the left
     */
    scrollItemsToLeft() {
      const { itemsListSlides } = this.$refs

      if (itemsListSlides) {
        jQuery(itemsListSlides).css('transform', 'translate3d(0, 0, 0)')
      }
    },

    /**
     * Create or reset the banner items flexslider
     */
    itemsSlider() {
      const { itemsList } = this.$refs

      if (typeof jQuery.flexslider === 'function') {
        const flexsliderData = jQuery(itemsList).data('flexslider')

        if (!flexsliderData) {
          // Initialize slider
          jQuery(itemsList).flexslider({
            animation: 'slide',
            animationLoop: false,
            controlNav: false,
            itemWidth: 96,
            itemMargin: 10,
            slideshow: false,
          })
        } else {
          // Reset the settings
          flexsliderData.setup()
        }
      }
    },
  },

  watch: {
    /**
     * @param {Array} items
     * @param {Array} prevItems
     */
    cartItems(items, prevItems) {
      const hasNewItems = items.some((item) => !prevItems.includes(item))

      this.$nextTick(() => {
        // Always reset the slider
        this.itemsSlider()

        if (hasNewItems) {
          this.scrollItemsToLeft()
        }
      })
    },

    /**
     * @param {Boolean} state
     */
    show(state) {
      setTimeout(() => {
        this.delayedShow = state
        // Reset slider
        this.itemsSlider()
      }, 200)
    },
  },

  mounted() {
    jQuery(() => {
      this.setPageContainerPadding()
      this.itemsSlider()

      // Trigger resize to re-init slider (weird bug)
      jQuery(window).trigger('resize')
    })
    jQuery(window).on('resize', () => {
      this.setPageContainerPadding()
      this.$nextTick(this.itemsSlider)
    })
    jQuery(document).on('scroll', this.setPageContainerPadding)
    jQuery(document).on('cart.requestStarted', this.onBeforeCartUpdate)
    jQuery(document).on('cart.requestComplete', this.onCartUpdate)
  },
})

Vue.component('product-bisn', {
  template: '#product-bisn',

  props: {
    customerId: [Number, String],
    email: String,
    productId: [Number, String],
    sku: String,
    hideAdditionalText: {
      type: Boolean,
      default: false,
    },
  },

  data() {
    return {
      apiUrl: '/apps/shop-manager/api',
      subscribed: false,
      loading: false,
    }
  },

  computed: {
    isLoggedIn() {
      return this.customerId > 0
    },
  },

  methods: {
    getRequestData() {
      const { shop } = Shopify
      const { customerId, email, productId, sku } = this.$props

      return {
        shop,
        customerId: `${customerId}`,
        email,
        productId: `${productId}`,
        sku,
        subscribe: this.subscribed,
      }
    },

    async sendPostRequest(url, data) {
      const rawResponse = await fetch(url, {
        method: 'POST',
        headers: {
          'Accept': 'application/json',
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(data),
      })

      return rawResponse.status === 200 && rawResponse.json()
    },

    async checkIfSubscribed() {
      this.loading = true

      const result = await this.sendPostRequest(
        `${this.apiUrl}/bisn/subscribed`,
        this.getRequestData(),
      )

      this.subscribed = !!result
      this.loading = false
    },

    async onCheckboxChange() {
      this.loading = true

      // Update state
      this.subscribed = !this.subscribed

      // Update subscription
      try {
        const result = await this.sendPostRequest(
          `${this.apiUrl}/bisn`,
          this.getRequestData(),
        )

        if (!result) throw new Error()
      } catch (e) {
        // Revert state
        this.subscribed = !this.subscribed
      }

      this.loading = false
    },
  },

  created() {
    this.checkIfSubscribed()
  },
})
</script>

    <script>
      const app = new Vue({
        el: '#app',

        data() {
          return {
            appData: APP_DATA,
            isCartModalOpen: false,
            isCartBannerShown: true,
            breakpoints: {
              MOBILE: 480,
              TABLET: 768,
              LAPTOP: 960,
              DESKTOP: 1200,
            },
            isMobileResolution: false,
            isTabletResolution: false,
            defaultQtyMinimum: 0,
            defaultQtyMaximum: 50,
            customerProductLimits: null,
          };
        },

        computed: {
          isDevDomain() {
            return this.appData.devDomains.includes(window.location.host)
          },
        },

        methods: {
          /**
           * @returns {Object[]}
           */
          getBoughtItemQtiesGroupedBySku() {
            const boughtItems = this.appData.boughtItems;
            const boughtItemQties = {};

            for (const { sku, qty } of boughtItems) {
              const oldQty = boughtItemQties[sku] || 0;

              boughtItemQties[sku] = oldQty + qty;
            }

            return boughtItemQties;
          },

          /**
           * Group GET request parameters as an object
           *
           * @returns {Object}
           */
          searchParams() {
            const paramsMap = window.location.search
              .replace('?', '')
              .split('&')
              .map(param => param.split('='));

            // Convert the params map to an object
            const params = {};

            for (const [key, value] of paramsMap) {
              params[key] = value;
            }

            return params;
          },

          /**
           * Set resolution booleans
           */
          checkResolutions() {
            const ww = window.outerWidth;

            this.isMobileResolution = ww <= this.breakpoints.MOBILE;
            this.isTabletResolution = ww <= this.breakpoints.TABLET;
          },

          /**
           * Helper method to fetch asset with typecheck
           */
          asset(key, fallback = null) {
            const asset = this.appData.assets[key];

            return typeof asset !== 'undefined' ? asset : fallback;
          },

          /**
           * Open the cart modal
           */
          openCart() {
            this.isCartModalOpen = true;
          },

          /**
           * Close the cart modal
           */
          closeCart() {
            this.isCartModalOpen = false;
          },

          /**
           * Show the cart banner
           */
          showCartBanner() {
            this.isCartBannerShown = true;
          },

          /**
           * Hide the cart banner
           */
          hideCartBanner() {
            this.isCartBannerShown = false;
          },

          /**
           * Helper method to format money
           *
           * @param {number} cents
           * @returns {string}
           */
          formatMoney(cents) {
            const money = Shopify.formatMoney(cents).split('.');
            const suffix  = money[1] === '00' ? '' : `.${money[1]}`;

            return `<span>${money[0]}</span><span class="small">${suffix}</span>`;
          },

          /**
           * Set-up all cart related actions
           */
          initCart() {
            // Initiate CartJS
            CartJS.init(this.appData.cart, {
              'requestBodyClass': 'loading',
              'moneyFormat': '${{amount}}',
              'moneyWithCurrencyFormat': '${{amount}} CAD'
            });

            // Open the cart if requested
            if (window.location.search === "?cart") {
              window.history.replaceState({}, null, window.location.pathname);
              this.openCart();
            }

            // Handle cart header menu item
            jQuery(document).on('click', '.site-header__cart-toggle', e => {
              e.preventDefault();
              this.openCart();
            });
          },

          setProductSku() {
            const { product } = this.appData;
            const { variant: variantId } = this.searchParams();
            const skuEl = document.querySelector('.product-current-sku');

            if (product && skuEl) {
              // Use the first variant by default
              let variant = product.variants[0];

              if (variantId) {
                // Find the variant based on the request parameter
                variant = product.variants.find(
                  variant => parseInt(variant.id) === parseInt(variantId)
                );
              }

              if (variant) {
                // Update the SKU
                let gen2 = new RegExp('-gen2', 'i');
                let standalone = new RegExp('-standalone', 'i');
                skuEl.innerText = variant.sku.replace(gen2, '').replace(standalone, '');
              }
            }
          },

          makeSlug(text) {
            return text.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/-$/, '').replace(/^-/, '');
          },

          getUrlParameter(sParam) {
              var sPageURL = decodeURIComponent(window.location.search.substring(1)),
              sURLVariables = sPageURL.split('&'),
              sParameterName,
              i;

              for (i = 0; i < sURLVariables.length; i++) {
              sParameterName = sURLVariables[i].split('=');

              if (sParameterName[0] === sParam) {
                  return sParameterName[1] === undefined ? true : sParameterName[1];
              }
              }
          },

          replaceUrlParam(e,r,a){
            var n=new RegExp("("+r+"=).*?(&|$)"),c=e;return c=e.search(n)>=0?e.replace(n,"$1"+a+"$2"):c+(c.indexOf("?")>0?"&":"?")+r+"="+a
          },

          /**
           * Handle [data-toggle] that toggles [data-toggle-id] elements
           */
          handleDataToggle(e) {
            const $action = jQuery(e.currentTarget);
            const targetId = $action.data('toggle');
            const $target = jQuery(`[data-toggle-id="${targetId}"]`);

            const toggleText = $action.attr('data-toggle-text');
            const toggleTextNew = $action.attr('data-toggle-text-new');
            const toggleTextTarget = $action.children('[data-toggle-text-target]').first();

            const forceOpen = typeof $action.data('toggle-open') !== 'undefined';
            const forceClose = typeof $action.data('toggle-close') !== 'undefined' && !forceOpen;

            e.preventDefault();

            if ($action.hasClass('active') || forceClose) {
              if (!forceOpen) {
                $target.removeClass('active');
                $target.stop(true).slideUp(300);
                $action.removeClass('active');
              }
            } else if (!$action.hasClass('active') || forceOpen) {
              if (!forceClose) {
                $target.addClass('active');
                $target.stop(true).slideDown(300);
                $action.addClass('active');
              }
            }

            if(!!toggleText && !!toggleTextNew && !!toggleTextTarget) {
              $action.hasClass('active') ? toggleTextTarget.text(toggleTextNew) : toggleTextTarget.text(toggleText);
            }

          },

          /**
           * Handle [data-scroll-to] that scrolls to [data-scroll-to-id] elements
           */
          handleDataScrollTo(e) {
            const $action = jQuery(e.currentTarget);
            const targetId = $action.data('scroll-to');
            const $target = jQuery(`[data-scroll-to-id="${targetId}"]`);

            jQuery('html, body').animate({
              scrollTop: $target.offset().top,
            }, 600);
          },

          /**
           * Fetch and parse a cart attribute
           *
           * @param {string} name
           * @returns {Object}
           */
          getCartAttribute(name) {
            const attributes = CartJS.getAttributes();
            const attribute = attributes && attributes[name] ? attributes[name] : '{}';

            return JSON.parse(attribute.replace(/=>/g, ':'));
          },

          /**
           * @param {boolean} forceUpdate
           * @returns {Promise<{ sku: string; max: string }[]>}
           */
          async getCustomerProductLimits(forceUpdate = false) {
            if (!this.customerProductLimits || forceUpdate) {
              const response = await fetch('/apps/shop-manager/api/product-limits', {
                method: 'POST',
                headers: {
                  Accept: 'application/json',
                  'Content-Type': 'application/json',
                },
                body: JSON.stringify({ shop: Shopify.shop }),
              })

              this.customerProductLimits = await response.json();
            }

            return this.customerProductLimits;
          },

          /**
           * Mobile screen VH fix
           */
          calcViewportHeight() {
            // First we get the viewport height and we multiple it by 1% to get a value for a vh unit
            let vh = window.innerHeight * 0.01

            // Then we set the value in the --vh custom property to the root of the document
            document.documentElement.style.setProperty('--vh', `${vh}px`)
          },

          redirectToNewStore() {
            window.location = "https://ca.store.ui.com/"
          },

          /**
           * Handle login referral redirect
           *
           * @returns {void}
           */
          handleLoginReferral() {
            if (!localStorage) {
              return
            }

            // If referral exists and not on login page
            const accountLoginRef = localStorage.getItem('account-login-ref')
            const isLoginPage = location.pathname === '/account/login'

            if (!accountLoginRef || isLoginPage) {
              return
            }

            // Remove the item
            localStorage.removeItem('account-login-ref')

            // Only if on account page (after login)
            const isAccountPage = location.pathname === '/account'

            if (!isAccountPage) {
              return
            }

            // Redirect to the referred url
            window.location = accountLoginRef
          },

          redirectToLogout() {
            if (this.isDevDomain) {
              window.location = '/account/logout'
            } else {
              window.location = `https://sso.ui.com/api/sso/v1/logout?redirect=https%3A%2F%2F${window.location.host}%2Faccount%2Flogout`
            }
          },

          redirectToLogin() {
            // Set a local storage referral url to current page
            if (localStorage) {
              localStorage.setItem('account-login-ref', location.href)
            }

            // Redirect
            location.href = '/account/login';
          },
        },

        created() {
          /*
          * #
          * #

          this.checkResolutions();
          this.initCart();

          * #
          * #
          */
        },

        mounted() {
          /**
           * For redirect view
           */
          jQuery(window).on('resize', () => {
            this.calcViewportHeight()
          });

          setTimeout(() => {
            this.redirectToNewStore()
          }, 2500)

          /*
          * #
          * #

          this.setProductSku();
          this.handleLoginReferral()
          this.calcViewportHeight()

          // Events
          jQuery(document).on('click', '[data-toggle]', this.handleDataToggle);
          jQuery(document).on('click', '[data-scroll-to]', this.handleDataScrollTo);
          jQuery(window).on('resize', () => {
            this.checkResolutions()
            this.calcViewportHeight()
          });

          // pushState event hack
          const pushState = history.pushState;
          history.pushState = (...args) => {
            pushState.apply(history, args);

            // Custom events
            this.setProductSku();
          };

          * #
          * #
          */

        },
      });
    </script>

  const carouselApp = {
    state: {
      currentSlide: 1,

      animationTimeout: null,
      isAnimating: false,
      touchstartX: 0,

    },

    // Get count of slides in carousel

      return this.elements.slides.childElementCount
    },

    // Get width of the overflow container in carousel
    getContainerWidth() {
      return this.elements.container.getBoundingClientRect().width
    },

    handleGesture() {
      const { touchendX, touchstartX } = this.state
      const touchSlideDistance = (touchstartX - touchendX)

      if (touchSlideDistance < -50) {
        this.slideLeft(true)
      } else if (touchSlideDistance > 50) {
        this.slideRight(true)
      } else {
        this.slideTo(this.state.currentSlide, true)
      }
    },

      const { currentSlide } = this.state
      const { slides } = this.elements

      if (clickEvent && clickEvent.stopPropagation) {
        clickEvent.stopPropagation()
      }

        this.slideTo(currentSlide - 1, cancelInterval)
      } else {
        this.slideTo(currentSlide)
      }
    },

    // Helper to slide one step right
    slideRight(cancelInterval = false, clickEvent) {
      const { currentSlide } = this.state
      const { slides } = this.elements

      if (clickEvent && clickEvent.stopPropagation) {
        clickEvent.stopPropagation()
      }

      if (currentSlide < this.getSlidesCount()) {
        this.slideTo(currentSlide + 1, cancelInterval)
      } else {
        this.slideTo(currentSlide)
      }
    },

    // Slide to slide number - (index 1)
    slideTo(slideNum, cancelInterval = false) {
      clearInterval(this.state.animationTimeout)
      this.state.isAnimating = true

      this.state.animationTimeout = setTimeout(() => {
        this.state.isAnimating = false
      }, 800)

      if (cancelInterval) {
        clearInterval(this.state.autoRotateInterval)
      }

      const { navLeft, navRight } = this.elements

      if (slideNum === 1) {
        navLeft.classList.add('disabled')
      } else {
        navLeft.classList.remove('disabled')
      }

      if (slideNum === this.getSlidesCount()) {
        navRight.classList.add('disabled')
      } else {
        navRight.classList.remove('disabled')
      }

      this.state.currentSlide = slideNum

      const leftPos = this.getContainerWidth() * (slideNum - 1)
      this.elements.slides.style.left = `${-leftPos}px`

      this.updateActiveThumb()
    },

    // Set the static width of slides
    setSlidesWidth() {
      const { slides, container } = this.elements
      const containerWidth = this.getContainerWidth()

      this.slideTo(1)

      slides.style.width = `${containerWidth}px`
    },

    // Update thumb elements to show current active slide
    updateActiveThumb() {
      const { currentSlide } = this.state
      const { footer } = this.elements

      let i = 0
      for (const el of footer.children) {
        if (i + 1 === currentSlide) {
          el.classList.add('active')
        } else {
          el.classList.remove('active')
        }

        i += 1
      }
    },

    initSlideThumbs() {
      const { footer } = this.elements
      const indicatorHTML = `<svg width="9" height="8" viewBox="0 0 9 8" fill="none" xmlns="http://www.w3.org/2000/svg"><circle cx="4.5" cy="4" r="4" fill="#C8CAD3"/></svg>`

      for (let i = 0; i < this.getSlidesCount(); i++) {
        const el = document.createElement('button')

        el.innerHTML = indicatorHTML
        el.id = `bannerCarouselThumb_${i + 1}`
        el.classList.add('banner-carousel__thumb')

        el.addEventListener('click', () => {
          this.slideTo(i + 1, true)
        })

        footer.appendChild(el)
      }

      this.updateActiveThumb()
    },

    initInterval() {
      this.state.autoRotateInterval = setInterval(() => {
        const { currentSlide, autoRotateDirection } = this.state
        const slidesCount = this.getSlidesCount()

        if (autoRotateDirection === 'right' && currentSlide < slidesCount) {
          this.slideRight()
        } else if (autoRotateDirection === 'right') {
          this.state.autoRotateDirection = 'left'
          this.slideLeft()
        } else if (autoRotateDirection === 'left' && currentSlide > 1) {
          this.slideLeft()
        } else if (autoRotateDirection === 'left') {
          this.state.autoRotateDirection = 'right'
          this.slideRight()
        }
      }, 6000)
    },

    init() {
      const {
        container,
        slides,
        navRight,
        navLeft,
        nav,
      } = this.elements

      this.setSlidesWidth()
      this.initSlideThumbs()
      this.slideTo(1)
      this.initInterval()

      window.addEventListener('resize', this.setSlidesWidth.bind(this))
      navRight.addEventListener('click', () => this.slideRight.bind(this)(true, event))
      navLeft.addEventListener('click', () => this.slideLeft.bind(this)(true, event))

      /**
       * Record touchstart X positon
       */
      container.addEventListener('touchstart', e => {
        if (this.state.isAnimating) {
          this.state.touchstartX = null
          return
        }

        slides.style.transition = 'none'
        this.state.touchstartX = e.changedTouches[0].screenX
      })

      /**
       * Visual "drag" animation
       */
      container.addEventListener('touchmove', e => {
        if (typeof this.state.touchstartX !== 'number') {
          return
        }

        const draggedDistance = e.changedTouches[0].screenX - this.state.touchstartX
        const currentSlideLeftPos = 0 - (this.getContainerWidth() * (this.state.currentSlide - 1))

        const draggedLeftPos = currentSlideLeftPos + (draggedDistance / 2)
        const minLeftPos = 0 - (this.getContainerWidth() * (this.getSlidesCount() - 1))

        if (draggedLeftPos < 0 && draggedLeftPos > minLeftPos) {
          slides.style.left = `${draggedLeftPos}px`
        }
      })

      /**
       * Record touchend X positon and check if valid gesture for carousel slide
       */
      container.addEventListener('touchend', e => {
        if (typeof this.state.touchstartX !== 'number') {
          return
        }

        slides.style.transition = 'left .8s'
        this.state.touchendX = e.changedTouches[0].screenX
        this.handleGesture()
      })

      nav.style.display = 'block'
    },
  }

  window.addEventListener('load', () => {
    carouselApp.init()
  })
</script>

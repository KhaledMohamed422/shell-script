Vue.component('cart-modal', {
  template: '#cart-modal',

  props: {
    open: Boolean,
    cartAccessories: Array,
  },

  delimiters: ['${', '}'],

  data() {
    return {
      loading: false,
      cart: CartJS.cart,
      tmpScrollTop: 0,
      mobileModes: {
        ITEMS: 0,
        SUGGESTED: 1,
      },
      mobileMode: 0,
      expandedProductIds: [],
      wasQuantityReduced: false,
      Shopify
    };
  },

  computed: {
    /**
     * @returns {Object}
     */
    customer() {
      return this.$root.appData.customer;
    },

    /**
     * Cart items shortcut, always returns an array
     *
     * @returns {Array}
     */
    cartItems() {
      return this.cart.items || [];
    },

    /**
     * Check if there are items in the cart
     *
     * @returns {boolean}
     */
    isCartEmpty() {
      return this.cartItems.length === 0;
    },

    /**
     * Get suggested products
     *
     * @returns {Array}
     */
    suggestedProducts() {
      // Return no products if the cart is empty - do not suggest anything
      if (this.isCartEmpty) {
        return [];
      }

      // Clone the accessories
      let accessories = this.cartAccessories.slice(0);

      // Sort based on the priority
      accessories = accessories.sort(
        (a, b) => parseFloat(b.data.priority || 1) - parseFloat(a.data.priority || 1),
      );

      return accessories;
    },

    /**
     * Get suggested product variants that are available
     *
     * @returns {Array}
     */
    suggestedVariants() {
      let variants = [];
      const maxVariantCount = 15;

      const checkView = viewData => {
        // Legacy boolean value
        if (!Array.isArray(viewData)) {
          return false;
        }

        // Ignored SKU not in cart
        if (viewData.some(sku => this.cartItems.some(item => item.sku === `!${sku}`))) {
          return false;
        }

        // Wildcard
        if (viewData.includes('*')) {
          return true;
        }

        // SKU in cart
        return viewData.some(sku => this.cartItems.some(item => item.sku === sku));
      };

      for (const product of this.suggestedProducts) {
        for (const variant of product.variants) {
          const isInCart = this.cartItems.some(
            item => item.sku === variant.sku,
          );

          // Add product object reference to the variant
          variant.product = product;

          // Add the variant to the list if not already in the cart
          if (!isInCart) {
            variants.push(variant);
          }
        }
      }

      // Hide variants that are not allowed for cart items
      variants = variants.filter(variant => {
        const data = variant.product.data || {};
        const dataFor = data.for || {};

        const checkFor = (dataKey, itemKey) => {
          if (dataFor[dataKey] && dataFor[dataKey].length) {
            let found = false;

            dataFor[dataKey].forEach(value => {
              if (this.cartItems.some(item => item[itemKey] === value)) {
                found = true;
                return;
              }
            });

            // Valid if targetted product is found in the cart
            return found;
          }

          // Valid by default
          return true;
        };

        // Check the country validity
        if (
          Array.isArray(data.countries) &&
          data.countries.length &&
          !data.countries.includes('US')
        ) {
          return false;
        }

        // Check for product match
        if (data.view && data.view.checkout && !checkView(data.view.checkout)) {
          return false;
        }

        // Lastly, check for targetted products
        return (
          checkFor('product-vendors', 'vendor') &&
          checkFor('product-types', 'product_type')
        );
      });

      // Limit variant count
      variants.splice(maxVariantCount);

      return variants;
    },

    /**
     * Check for suggested product count
     *
     * @returns {boolean}
     */
    hasSuggestedProducts() {
      return this.suggestedProducts.length > 0;
    },

    /**
     * Check for suggested product variant count
     *
     * @returns {boolean}
     */
    hasSuggestedVariants() {
      return this.suggestedVariants.length > 0;
    },

    /**
     * Check if accented sections should be shown
     *
     * @returns {boolean}
     */
    showAccented() {
      return (
        !this.$root.isMobileResolution ||
        this.mobileMode === this.mobileModes.ITEMS
      );
    },
  },

  methods: {
    /**
     * Toggle suggested product variant collapsing
     *
     * @param {number} productId
     */
    toggleSuggestedProduct(productId) {
      const expandedProductIndex = this.expandedProductIds.indexOf(productId);

      if (expandedProductIndex === -1) {
        // Expand
        this.expandedProductIds.push(productId);
      } else {
        // Collapse
        this.expandedProductIds.splice(expandedProductIndex, 1);
      }
    },

    /**
     * Check if the product is expanded
     *
     * @param {number} productId
     * @returns {Boolean}
     */
    suggestedProductExpanded(productId) {
      return this.expandedProductIds.includes(productId);
    },

    /**
     * Check to see if the suggested variant can be shown
     *
     * @param {Object} variant
     * @returns {boolean}
     */
    canShowSuggestedVariant(variant) {
      return this.suggestedVariants.some(({ sku }) => sku === variant.sku);

    /**

     * @param {Object} product
     * @returns {Array}

    productSuggestedVariants(product) {
      return product.variants.filter(this.canShowSuggestedVariant);
    },

    /**
     * Set the current mobile mode

     * @param {number} mode

    setMobileMode(mode) {

    },

    /**
     * Parse the variant and build a title

     * @param {Object} variant
     * @returns {string}

    getVariantTitle(variant) {
      let title = variant.product.title;

      if (variant.title !== 'Default Title') {
        title += ` - ${variant.title}`;

      return title;

    /**
     * Send a custom close event for the parent to handle
     */
    closeModal() {
      this.$emit('close');
    },

    /**
     * Enable loading when the cart is about to update
     */
    onBeforeCartUpdate() {
      this.loading = true;
    },

    /**
     * Update component data on cart update
     *
     * @param {Object} event
     * @param {Object} cart
     */
    onCartUpdate(event, cart) {
      // Update the cart data
      this.cart = cart;
      this.loading = false;
    },

    /**
     * Validate cart quantities
     * Attribute are set in vue-product-cart-relations
     */
    async checkCartProductQuantities() {
      const { items } = this.cart;
      const limitsAttribute = this.$root.getCartAttribute('quantity-limits');
      const kitsAttribute = this.$root.getCartAttribute('quantity-kits');
      const hddsAttribute = this.$root.getCartAttribute('quantity-hdds');

      const itemQuantities = {};
      const debugQuantityChanges = false;
      const setItemQuantity = (itemId, quantity, reason = '') => {
        if (debugQuantityChanges) {
          console.log(`${itemId} was set as ${quantity}`, reason ? `: ${reason}` : '');
        }
        itemQuantities[itemId] = quantity;
      };

      // Manage HDDs
      const cartHddIdMaxQties = {};
      const allCartHddIds = [];

      for (const [dependantProductId, hdds] of Object.entries(hddsAttribute)) {
        const dependantCartItem = items.find(
          ({ product_id: productId }) =>
            `${productId}` === `${dependantProductId}`,
        );
        const hddSkus = Object.values(hdds).map(({ sku }) => sku);

        items
          .filter(({ sku }) => hddSkus.includes(sku))
          .forEach(({ id, sku }) => {
            // Store the HDD id
            if (!allCartHddIds.includes(id)) {
              allCartHddIds.push(id);
            }

            // Set the max allowed quantity if possible
            if (dependantCartItem) {
              const hdd = Object.values(hdds).find(hdd => hdd.sku === sku);
              const prevQty = parseInt(cartHddIdMaxQties[id]) || 0;
              const dependantQty = parseInt(dependantCartItem.quantity);
              const ratio = (hdd && parseInt(hdd.ratio)) || 1;

              cartHddIdMaxQties[id] = prevQty + dependantQty * ratio;
            }
          });
      }

      for (const cartHddId of allCartHddIds) {
        const cartHdd = items.find(({ id }) => `${id}` === `${cartHddId}`);
        const cartHddMaxQty = parseInt(cartHddIdMaxQties[cartHddId]) || 0;

        if (cartHdd && cartHdd.quantity > cartHddMaxQty) {
          // Limit the allowed max quantity
          setItemQuantity(cartHddId, cartHddMaxQty, 'HDD max limit');
        }
      }

      // Manage kits
      for (const [parentProductId, kitItems] of Object.entries(kitsAttribute)) {
        const parentCartItem = items.find(
          ({ product_id: productId }) =>
            `${productId}` === `${parentProductId}`,
        );
        const parentProductLimits = limitsAttribute[parentProductId];

        if (parentCartItem) {
          for (const kitItem of Object.values(kitItems)) {
            // Add the kit item to the cart if not there already
            if (!items.some(({ id }) => `${id}` === `${kitItem.itemId}`)) {
              const quantity = parseInt(kitItem.quantity) || 1;

              setItemQuantity(kitItem.itemId, quantity * parentCartItem.quantity, 'Kit item');
            }
          }

          // Remove the parent item from cart
          setItemQuantity(parentCartItem.id, 0, 'Kit parent item');
        } else {
          // Check and manage the cart kit items
          const kitItemIds = Object.values(kitItems).map(
            ({ itemId }) => `${itemId}`,
          );
          const kitCartItems = items.filter(({ id }) =>
            kitItemIds.includes(`${id}`),
          );

          if (kitCartItems.length < kitItemIds.length) {
            // Remove dangling kit items
            for (const kitItem of Object.values(kitItems)) {
              if (
                kitCartItems.some(({ id }) => `${id}` === `${kitItem.itemId}`)
              ) {
                setItemQuantity(kitItem.itemId, 0, 'Dangling kit item');
              }
            }
          } else {
            // Manage kit item quantity linking
            const kitCartQties = kitCartItems.map(({ quantity }) =>
              parseInt(quantity),
            );
            const kitCartQtiesUnique = [...new Set(kitCartQties)];

            if (kitCartQtiesUnique.length > 1) {
              const quantity = this.wasQuantityReduced
                ? Math.min(...kitCartQtiesUnique)
                : Math.max(...kitCartQtiesUnique);

              for (const kitItemId of kitItemIds) {
                setItemQuantity(kitItemId, quantity, 'Equalize kit item quantities');
              }
            }
          }
        }

        // Adjust the quantity limits based on the parent
        if (parentProductLimits) {
          const { minimum, maximum } = parentProductLimits;

          for (const { itemId } of Object.values(kitItems)) {
            const quantity = itemQuantities[itemId];

            if (maximum && quantity > maximum) {
              setItemQuantity(itemId, maximum, 'Max allowed parent qty');
            } else if (minimum && quantity < minimum) {
              setItemQuantity(itemId, minimum, 'Min allowed parent qty');
            }
          }
        }
      }

      // Manage limits - has to be the last one
      for (const [limitProductId, limits] of Object.entries(limitsAttribute)) {
        const minimum = parseInt(limits.minimum) || this.$root.defaultQtyMinimum;
        const maximum = parseInt(limits.maximum) || this.$root.defaultQtyMaximum;
        const multiple = parseInt(limits.multiple) || null;

        for (const item of items) {
          const { id, product_id: productId } = item;
          let { quantity } = item

          if (`${productId}` === `${limitProductId}`) {
            // Enforce step - has to be the first one
            if (multiple !== null && multiple >= 0 && quantity % multiple !== 0) {
              const diff = quantity % multiple;
              const additive = this.wasQuantityReduced ? -diff : multiple - diff;

              quantity += additive;
              setItemQuantity(id, quantity, 'Multi step quantity');
            }

            // Enforce max quantity
            if (maximum !== null && maximum >= 0 && quantity > maximum) {
              setItemQuantity(id, maximum, 'Max allowed qty');
            }

            // Enforce min quantity
            if (minimum !== null && minimum >= 0 && quantity < minimum) {
              // Allow setting it to 0 always
              setItemQuantity(id, quantity > 0 ? minimum : 0, 'Min allowed quantity');
            }
          }
        }
      }

      // Per-customer limits
      const customerProductLimits = await this.$root.getCustomerProductLimits();

      if (customerProductLimits) {
        const notLoggedIn = !(this.customer.email && this.customer.email.length);
        const boughtItemQties = this.$root.getBoughtItemQtiesGroupedBySku();

        for (const item of items) {
          const { id, quantity } = item;
          const variantMaxQtyLimit = customerProductLimits.find(
            limit => limit.sku.toLowerCase() === item.sku.toLowerCase()
          );

          if (!variantMaxQtyLimit) {
            continue;
          }

          if (notLoggedIn) {
            // Set the item quantity to 0 due to requiring a login
            setItemQuantity(id, 0, 'Customer product limits requires logging in');
            continue;
          }

          const boughtQty = boughtItemQties[item.sku] || 0;
          const customerAllowedQty = variantMaxQtyLimit.max - boughtQty;

          if (quantity > customerAllowedQty) {
            // Update the item quantity to the allowed limit
            const finalItemQty = customerAllowedQty >= 0 ? customerAllowedQty : 0;

            setItemQuantity(id, finalItemQty, 'Customer product limits max allowed');
          }
        }
      }

      // Update quantities
      if (Object.values(itemQuantities).length) {
        // If any product is sold out or fails, all updates are cancelled so it's individual updates
        for (const [itemId, quantity] of Object.entries(itemQuantities)) {
          if (/^[0-9]+$/.test(quantity)) {
            CartJS.updateItemQuantitiesById({ [itemId]: quantity });
          } else {
            console.error(`Could not update ${itemId} to ${quantity}`);
          }
        }
      }
    },

    /**
     * Redirect to checkout
     */
    navigateToCheckout() {
      if (!this.isCartEmpty) {
        this.loading = true;
        window.location = `/checkout?locale=${app.currentLang}`;
      }
    },

    /**
     * Scroll the items list to the top
     */
    scrollItemsToTop() {
      const { itemsList } = this.$refs;

      if (itemsList) {
        jQuery(itemsList).animate({ scrollTop: 0 }, 300);
      }
    },

    /**
     * Update target element text to match cart total item count
     */
    updateCartItemCountEls() {
      const { item_count: count } = this.cart;
      const countEls = document.querySelectorAll('.cart-item-count');

      if (countEls) {
        countEls.forEach(el => {
          el.innerText = count;
        });
      }
    },

    removeSoldOutItems() {
      const [ lastAddedItem ] = this.cart.items
      const isSoldOutCheck = product => product.tags.split(', ').some(tag => tag === 'sold-out')

      const removeLastAddedItem = () => {
        CartJS.removeItemById(lastAddedItem.id)
      }

      if(lastAddedItem && lastAddedItem.handle) {
        fetch(`/products/${lastAddedItem.handle}.json`)
          .then(data => data.json())
          .then(({ product }) => isSoldOutCheck(product))
          .then(isSoldOut => isSoldOut && removeLastAddedItem())
      }
    },

    onCartOpen() {
      const { id, items } = this.cart

      // Track
      Segment.track('Cart Viewed', {
        cart_id: id,
        products: items.map((item, i) => ({
          product_id: item.product_id,
          sku: item.sku,
          category: item.product_type,
          name: item.product_title,
          brand: item.vendor,
          variant: item.variant_id,
          price: Segment.centsToMoney(item.price),
          quantity: item.quantity,
          coupon: '',
          position: i,
          url: `https://${window.location.host}${item.url}`,
          image_url: item.image,
        })),
      })

      // Disable body scrolling
      this.tmpScrollTop = jQuery(window).scrollTop()
      jQuery('body').css('overflow', 'hidden')

      // Reset the mobile mode to items
      this.setMobileMode(this.mobileModes.ITEMS)

      // Reset scroll
      this.scrollItemsToTop()
    },

    onCartClose() {
      // Enable body scrolling
      jQuery('body').css('overflow', 'auto')
      jQuery(window).scrollTop(this.tmpScrollTop)
    },

    onItemAdd(item, itemIndex) {
      // Track
      Segment.track('Product Added', {
        cart_id: '',
        product_id: item.product_id,
        sku: item.sku,
        category: item.product_type,
        name: item.product_title,
        brand: item.vendor,
        variant: item.variant_id,
        price: Segment.centsToMoney(item.price),
        quantity: item.quantity,
        coupon: '',
        position: itemIndex,
        url: `https://${window.location.host}${item.url}`,
        image_url: item.image,
      })
    },

    onItemRemove(item, itemIndex) {
      // Track
      Segment.track('Product Removed', {
        cart_id: '',
        product_id: item.product_id,
        sku: item.sku,
        category: item.product_type,
        name: item.product_title,
        brand: item.vendor,
        variant: item.variant_id,
        price: Segment.centsToMoney(item.price),
        quantity: item.quantity,
        coupon: '',
        position: itemIndex,
        url: `https://${window.location.host}${item.url}`,
        image_url: item.image,
      })
    },
  },

  watch: {
    /**
     * @param {Array} items
     * @param {Array} prevItems
     */
    cartItems(items, prevItems) {
      // Update counts
      this.updateCartItemCountEls()

      // Update quantity change check
      const totalQty = items
        .map(({ quantity }) => parseInt(quantity))
        .reduce((a, b) => a + b, 0)
      const prevTotalQty = prevItems
        .map(({ quantity }) => parseInt(quantity))
        .reduce((a, b) => a + b, 0)

      this.wasQuantityReduced = totalQty < prevTotalQty

      // On items change
      const addedItems = items.filter(
        (item) => !prevItems.map((x) => x.id).includes(item.id),
      )
      const removedItems = prevItems.filter(
        (item) => !items.map((x) => x.id).includes(item.id),
      )

      for (const item of addedItems) {
        const itemIndex = items.findIndex(({ id }) => id === item.id)

        this.onItemAdd(item, itemIndex)
      }

      for (const item of removedItems) {
        const itemIndex = prevItems.findIndex(({ id }) => id === item.id)

        this.onItemRemove(item, itemIndex)
      }

      if (addedItems.length) {
        // Scroll items to top if added
        this.scrollItemsToTop()

        // Remove item if it has "sold-out" tag
        this.removeSoldOutItems()
      }

      // Check cart product links and limits
      this.checkCartProductQuantities()
    },

    /**
     * @param {boolean} state
     */
    open(state) {
      if (state) {
        this.onCartOpen()
      } else {
        this.onCartClose()
      }
    },

    /**
     * @param {Array} variants
     */
    suggestedVariants(variants) {
      if (!variants.length) {
        // Switch to items mode when no suggested items remain
        this.setMobileMode(this.mobileModes.ITEMS);
      }
    },
  },

  mounted() {
    // Body ready actions
    jQuery(() => {
      this.updateCartItemCountEls();
    });

    // Cart events
    jQuery(document).on('cart.requestStarted', this.onBeforeCartUpdate);
    jQuery(document).on('cart.requestComplete', this.onCartUpdate);
  },

  beforeDestroy() {
    jQuery(document).off('cart.requestStarted', this.onBeforeCartUpdate);
    jQuery(document).off('cart.requestComplete', this.onCartUpdate);
  },
});
</script>

     *

     *

      if (itemsListSlides) {
        jQuery(itemsListSlides).css('transform', 'translate3d(0, 0, 0)')
      }

    itemsSlider() {
      const { itemsList } = this.$refs

<script>
Vue.component('cart-product-item', {
  template: '#cart-product-item',

  props: {
    id: Number,
    title: String,
    image: String,
    price: Number,
    quantity: Number,
    disabled: Boolean,
    url: String,
    hideDetails: Boolean,
  },

  data() {
    return {
      limit: {},
      isLimitEnabled: false,
    };
  },

  delimiters: ['${', '}'],

  computed: {
    /**
     * Base price multiplied by the quantity
     * Shows the base price for both 0 and 1 quantities
     *
     * @returns {number}
     */
    finalPrice() {
      return this.quantity > 0 ? this.price * this.quantity : this.price;
    },

    /**
     * Get the item product ID
     *
     * @returns {number}
     */
    productId() {
      const item = CartJS.cart.items.find(item => item.id === this.id);

      return item && parseInt(item.product_id);
    },

    /**
     * Check if the quantity can be increased
     *
     * @returns {boolean}
     */
    canIncreaseQty() {
      if (this.localQuantity >= 9999) {
        return false;
      };

      if (!this.isLimitEnabled) {
        return true;
      };

      return this.localQuantity < this.limit.max;
    },

    /**
     * Check if the quantity can be decreased
     *
     * @returns {boolean}
     */
    canDecreaseQty() {
      return this.localQuantity > 0
    },
  },

  methods: {
    /**
     * Update the product limits data
     */
    async setLimit () {
      const limits = await this.$root.getCustomerProductLimits()
      const limit = limits.some(limit => limit.variantId === this.id)

      if (limit) {
        this.limit = limit
        this.isLimitEnabled = true
      }
    },

    /**
     * Update cart item quantity
     *
     * @param {number} itemId
     * @param {number} quantity
     */
    updateQuantity(id, quantity) {
      if (!this.disabled) {
        CartJS.updateItemQuantitiesById({ [id]: quantity });
      }
    },

    /**
     * Helper method to increase cart item quantity
     */
    increaseQty() {
      this.canIncreaseQty && this.updateQuantity(this.id, this.quantity + 1);
    },

    /**
     * Helper method to decrease cart item quantity
     */
    decreaseQty() {
      this.updateQuantity(this.id, this.quantity - 1);
    },

    /**
     * Update quantity from input value
     *
     * @param {Event} event
     */
    updateQtyFromInput(event) {
      const { value } = event.target;
      const newQuantity = parseInt(value);

      if (newQuantity >= 0) {
        // Update the value if it's changed
        if (newQuantity !== this.quantity) {
          this.updateQuantity(this.id, newQuantity);
        }
      } else {
        // Reset the initial quantity
        event.target.value = this.quantity;
      }
    },
  },

  created() {
    this.setLimit()
  },
});
</script>

Vue.component('product-bisn', {
  template: '#product-bisn',

  delimiters: ['${', '}'],

  props: {
    customerId: [Number, String],
    email: String,
    productId: [Number, String],
    sku: String,
    hideAdditionalText: {
      type: Boolean,
      default: false,
    },
  },

  data() {
    return {
      t: app.t,
      apiUrl: '/apps/shop-manager/api',
      subscribed: false,
      loading: false,
    }
  },

  computed: {
    isLoggedIn() {
      return this.customerId > 0
    },
  },

  methods: {
    getRequestData() {
      const { shop } = Shopify
      const { customerId, email, productId, sku } = this.$props

      return {
        shop,
        customerId: `${customerId}`,
        email,
        productId: `${productId}`,
        sku,
        subscribe: this.subscribed,
      }
    },

    async sendPostRequest(url, data) {
      const rawResponse = await fetch(url, {
        method: 'POST',
        headers: {
          'Accept': 'application/json',
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(data),
      })

      return rawResponse.status === 200 && rawResponse.json()
    },

    async checkIfSubscribed() {
      this.loading = true

      const result = await this.sendPostRequest(
        `${this.apiUrl}/bisn/subscribed`,
        this.getRequestData(),
      )

      this.subscribed = !!result
      this.loading = false
    },

    async onCheckboxChange() {
      this.loading = true

      // Update state
      this.subscribed = !this.subscribed

      // Update subscription
      try {
        const result = await this.sendPostRequest(
          `${this.apiUrl}/bisn`,
          this.getRequestData(),
        )

        if (!result) throw new Error()
      } catch (e) {
        // Revert state
        this.subscribed = !this.subscribed
      }

      this.loading = false
    },
  },

  created() {
    this.checkIfSubscribed()
  },
})

  computed: {
    hasCurrencySelect() {
      const shopOrigins = ['store-ui-europe.myshopify.com']

      return this.shop !== null && shopOrigins.includes(this.shop.origin)
    },

    currencySelectCurrencies() {
      const result = []

      if (this.hasCurrencySelect) {
        for (const { currency } of this.shop.countries) {
          if (!result.includes(currency)) {
            result.push(currency)
          }
        }
      }

      return result
    },
  },

  methods: {
    hasCookie() {
      return typeof jQuery.cookie(this.cookieKey) !== 'undefined'
    },

    setCookie(closeAction = false) {
      jQuery.cookie(this.cookieKey, closeAction ? '0' : '1', {
        expires: 30,
        path: '/',
      })
    },

    close() {
      // Set the cookie
      this.setCookie()

      // Hide the modal
      this.canShow = false
    },

    openShop() {
      if (!this.shop) {
        return
      }

      // Set the cookie
      this.setCookie()

      // Redirect
      const shopUrlParams = this.hasCurrencySelect
        ? '?' +
          [
            `refCountry=${this.selectedCountryCode}`,
            `refCurrency=${this.selectedCurrencyCode}`,
          ].join('&')
        : ''

      window.location = `${this.shop.url}${shopUrlParams}`
    },

    async initGeoCountry() {
      if (this.geoCountry) {
        return
      }

      try {
        const url = 'https://www.ui.com/api/geoip'
        const res = await fetch(url, { method: 'POST' })
        const { country_code: code, country_name: name } = await res.json()

        // Set the country as the selected country
        this.selectedCountryCode = code

        this.geoCountry = { code, name }
      } catch (e) {
        console.error('Failed to get country', e)
      }
    },

    initShop() {
      if (this.shop || !this.geoCountry) {
        return
      }

      for (const shop of this.shops) {
        const hasCountry = shop.countries.some(
          ({ code }) => code === this.geoCountry.code,
        )

        if (hasCountry) {
          this.shop = shop
          break
        }
      }
    },

    handleRefParams() {
      const { origin, searchParams } = new URL(location.href)
      let reloadPage = false

      // Set country
      const refCountry = searchParams.get('refCountry')

      if (refCountry) {
        jQuery.cookie('site-country', refCountry, {
          path: '/',
        })
        reloadPage = true
      }

      // Set currency
      const refCurrency = searchParams.get('refCurrency')

      if (refCurrency) {
        jQuery.cookie('site-currency', refCurrency, {
          path: '/',
        })
        reloadPage = true
      }

      // Reload the page to update content and remove parameters
      if (reloadPage) {
        window.location = origin
      }
    },
  },

  async mounted() {
    // Init data
    await this.initGeoCountry()
    this.initShop(this.geoCountry)

    jQuery(() => {
      // Handle ref parameters
      this.handleRefParams()

      // Check if modal can be shown
      this.canShow =
        !this.hasCookie() &&
        this.geoCountry &&
        this.shop &&
        this.shop.origin !== this.currentShopOrigin
    })
  },

<script data-ot-ignore type="text/javascript">

    const NOTES_APP = new Vue({
        el: '#notesApp',

        delimiters: ['${', '}'],

        data() {
            return {
                apiUrl: '/apps/shop-manager/api/notes',
                notes: [],
                marginBottom: 16,
            }
        },


        mounted() {
            this.fetchNotes();
            window.addEventListener('resize', this.handleWrapperSize);
        },

        computed: {
            shopOrigin() {
                return window.Shopify.Checkout
                    ? window.Shopify.Checkout.apiHost
                    : window.Shopify.shop
            },
            hasNotes() {
                return this.notes.length !== 0;
            }
        },

        methods: {
            async fetchNotes() {
                const rawResponse = await fetch(`${this.apiUrl}?shop=${this.shopOrigin}`);
                const response = rawResponse.status === 200 && await rawResponse.json();
                const currentDate = new Date();
                let notes = [];

                if (response && response.success) {
                    response.data.forEach(note => {
                        let hide = false;

                        if (typeof $.cookie === 'function') {
                            const noteCookie = $.cookie('comNote__' + note.uid);

                            if (noteCookie && noteCookie.substring(0, 7) === 'clicked') {
                                let dateCookieCreated = noteCookie.substring(9, noteCookie.length);
                                dateCookieCreated = new Date(dateCookieCreated)

                                //delete cookie if note updateed
                                const deleteCookie = dateCookieCreated.getTime() <= new Date(note.updatedAt).getTime();
                                if (deleteCookie) {
                                    $.removeCookie('comNote__' + note.uid);
                                } else {
                                    hide = true;
                                }
                            }
                        }

                        if (!hide && note.active && (currentDate.getTime() <= new Date(note.endDate).getTime() && currentDate.getTime() >= new Date(note.startDate).getTime())) {
                            notes.push(note);
                        }
                    });
                }
                this.notes = notes;
            },
            showNote(element) {
                element.show();
                const $wrapperHeight = element.find('.comNote__wrapper-inner').outerHeight(true);

                element.animate({
                    height: $wrapperHeight + this.marginBottom
                }, 500);
            },
            hideNote(id, duration, index) {
                const $el = $('#comNote__' + id);
                const $noteHeight = $el.outerHeight(true)

                const $wrapper = $('#page__notes-wrapper');
                const $wrapperHeight = $wrapper.outerHeight(true);

                this.notes.splice(index, 1);

                $wrapper.animate({
                    height: $wrapperHeight - $noteHeight
                }, 500);

                const currentDate = new Date();
                $.cookie('comNote__' + id, 'clicked__' + currentDate.toISOString(), {
                    expires: duration,
                    path: '/'
                });
            },
            handleWrapperSize() {
                if (this.hasNotes) {
                    const wrapper = this.$refs.comNote__wrapper;
                    const wrapperRec = wrapper.getBoundingClientRect()
                    const innerWrapper = this.$refs.comNote__wrapperInner;
                    const innerWrapperRec = innerWrapper.getBoundingClientRect()

                    if (innerWrapperRec.height > (wrapperRec.height - this.marginBottom)) {
                        wrapper.style.height = innerWrapperRec.height + this.marginBottom + 'px';
                    } else if (innerWrapperRec.height < (wrapperRec.height - this.marginBottom)) {
                        wrapper.style.height = innerWrapperRec.height + this.marginBottom + 'px';
                    }
                }
            }
        },
        watch: {
            notes(notes) {
                if (notes.length !== 0) {
                    setTimeout(() => {
                        const el = $('#page__notes-wrapper');
                        this.showNote(el);
                    }, 500);
                }
            },
        },
    })
</script>

new Vue({
  el: '#header-mobile-menu',

  delimiters: ['${', '}'],

  data() {
    return {
      show: false,
      tmpScrollTop: 0,
      menuData: HEADER_MENU_DATA,
      selectedMenuItem: null,
    };
  },

  computed: {
    selectedMenuItemType() {
      let type = '';

      if (this.selectedMenuItem) {
        const { url } = this.selectedMenuItem;

        if (url.includes('unifi')) {
          type = 'unifi';
        } else if (url.includes('operator')) {
          type = 'operator';
        } else if (url.includes('early-access')) {
          type = 'beta';
        }
      }

      return type;
    },

    selectedMenuItemIndex() {
      if (this.selectedMenuItem) {
        const { title, url } = this.selectedMenuItem;

        switch (this.selectedMenuItemType) {
          case 'unifi':
            return this.menuData.unifi.findIndex(item => item.title === title && item.url === url);

          case 'operator':
            return this.menuData.operator.findIndex(item => item.title === title && item.url === url);
        }
      }
      return -1;
    },

    selectedMenuItemTitleContent() {
      const { operatorMenuImages } = this.menuData;
      let content = '';

      if (this.selectedMenuItem && operatorMenuImages) {
        const { title, url } = this.selectedMenuItem;

        switch (this.selectedMenuItemType) {
          case 'operator':
            // Set the SVG as the content
            content = operatorMenuImages[this.selectedMenuItemIndex];
            break;

          default:
            // Set the title by default
            content = title;
        }
      }

      return content;
    },

    activeMenuTitle() {
        let unifiItem = this.menuData.unifi.filter(item => {
          return item.active
        })
  
        // let operatorItem =  this.menuData.operator.filter(item => {
        //   return item.active
        // })
        
        if(unifiItem[0]) {
          return unifiItem[0].title
  
        // } else if(operatorItem[0]) {
        //   return operatorItem[0].title
  
        } else {
          return false
        }
      }
  },

  methods: {
    /**
     * @param {string?} prefix
     * @returns {string}
     */
    selectedMenuItemClass(prefix = '') {
      const className = this.selectedMenuItemType || 'default';

      return `${prefix}${className}`;
    },

    /**
     * @param {Event} e
     */
    toggleMenu(e) {
      e.preventDefault();

      this.show = !this.show;
    },

    adjustMenuStyle() {
      const headerEl = document.querySelector('.comGlobalHeader');
      const menuEl = this.$refs.menu;

      if (menuEl) {
        // menuEl.style.top = `${headerEl.offsetHeight}px`;
        menuEl.style.height = `${window.innerHeight - headerEl.offsetHeight}px`;
      }
    },

    /**
     * @param {Object} item
     */
    menuItemClick(item) {
      if (item.links.length) {
        // Set the item as active menu item
        this.selectedMenuItem = item;
      } else {
        // Navigate to the page
        window.location = item.url;
      }
    },

    /**
     * @param {Event} e
     */
    deselectMenuItem(e) {
      e.preventDefault();

      this.selectedMenuItem = null;
    },

    /**
     * Force hide the cart banner
     * Using direct element manipulation is a bad practice,
     * however, there is no state management yet to do it the right way
     *
     * @param {boolean} state
     */
    toggleCartBanner(state) {
      // Use the outer wrappper to not interfere with the main element styles
      const cartBannerEl = document.querySelector('.cartBanner__outer');

      if (cartBannerEl) {
        cartBannerEl.style.display = !state ? 'none' : 'block';
      }
    }
  },

  watch: {
    /**
     * @param {boolean} state
     */
    show(state) {
      if (state) {
        // Scroll to the top of the page
        jQuery('html, body').scrollTop(0);

        // Disable body scrolling
        this.tmpScrollTop = jQuery(window).scrollTop();
        jQuery('body').css('overflow', 'hidden');

        // Hide the banner
        this.toggleCartBanner(false);
      } else {
        // Enable body scrolling
        jQuery('body').css('overflow', 'auto');
        jQuery(window).scrollTop(this.tmpScrollTop);

        // Show the banner
        this.toggleCartBanner(true);
      }
    },
  },

  mounted() {
    // Body ready actions
    jQuery(() => {
      this.adjustMenuStyle();
    })

    // Events
    window.addEventListener('resize', this.adjustMenuStyle);
  },

  beforeDestroy() {
    window.removeEventListener('resize', this.adjustMenuStyle);
  },
});

    <script type='text/javascript'>
      // On page load set the correct size of the iframe to account for either CWAE/Contact Us or Session Continuity.
      function helpButtonLoaded() {
        var helpButton = document.querySelector('.helpButton');
        if (!helpButton) {
          window.setTimeout(helpButtonLoaded, 500);
          return;
        }
        var height = helpButton.clientHeight;
        var width = helpButton.clientWidth;
        window.parent.postMessage({
          frameHeight: height,
          frameWidth: width
        }, '*'); // Used a * but for security reasons you can plug in the domain name.
      }
      window.onload = helpButtonLoaded();

      window.addEventListener('message', (e) => {
        const type = e.data && e.data.type;

        if (type === 'preset-user-data') {
          const payload = e.data.data;

          embedded_svc.settings.prepopulatedPrechatFields = { ...payload };
        }
      });

      var initESW = function(gslbBaseURL) {
        embedded_svc.settings.displayHelpButton = true; // Or false
        embedded_svc.settings.language = 'en-US'; // For example, enter 'en' or 'en-US'
        embedded_svc.settings.devMode = false;
        // Have set the widget height and width as a property on the object
        embedded_svc.settings.widgetWidth = 320;
        embedded_svc.settings.widgetHeight = 560;
        embedded_svc.settings.enabledFeatures = ['LiveAgent'];
        embedded_svc.settings.entryFeature = 'LiveAgent';
        embedded_svc.settings.offlineSupportMinimizedText = 'Submit a ticket';

        embedded_svc.init(
          'https://verygoodsecurity.my.salesforce.com',
          'https://support.verygoodsecurity.com/',
          gslbBaseURL,
          '00D1I000002Jf6c',
          'Chat_Agents_Vault',
          {
            baseLiveAgentContentURL: 'https://c.la2-c1-ord.salesforceliveagent.com/content',
            deploymentId: '5721I000000QWTh',
            buttonId: '5731I000000QT6x',
            baseLiveAgentURL: 'https://d.la2-c1-ord.salesforceliveagent.com/chat',
            eswLiveAgentDevName: 'Chat_Agents_Vault',
            isOfflineSupportEnabled: true
          }
        );

        embedded_svc.addEventHandler('afterMinimize', function() {
          var sidebar = document.querySelector('.embeddedServiceSidebarMinimizedDefaultUI');
          var height = sidebar.clientHeight;
          var width = sidebar.clientWidth;
          window.parent.postMessage({
            frameHeight: height,
            frameWidth: width
          }, '*');
        });

        embedded_svc.addEventHandler('afterMaximize', function() {
          var container = document.querySelector('.dockableContainer');
          var height = container.clientHeight;
          var width = container.clientWidth;

          window.parent.postMessage({
            frameHeight: embedded_svc.settings.widgetHeight,
            frameWidth: embedded_svc.settings.widgetWidth
          }, '*');

          // Hide WebEmail field
          const suppliedEmailFieldLabel = document.querySelector('label[for="SuppliedEmail"]');
          const suppliedEmailField = document.getElementById('SuppliedEmail');

          if (suppliedEmailFieldLabel && suppliedEmailField) {
            suppliedEmailFieldLabel.style.display = 'none';
            suppliedEmailField.style.display = 'none';
          }
        });

        embedded_svc.addEventHandler('afterDestroy', function() {
          var height = document.querySelector('.helpButton').clientHeight;
          var width = document.querySelector('.helpButton').clientWidth;
          window.parent.postMessage({
            frameHeight: height,
            frameWidth: width
          }, '*');
        });
      };

      if (!window.embedded_svc) {
        var s = document.createElement('script');
        s.setAttribute('src', 'https://verygoodsecurity.my.salesforce.com/embeddedservice/5.0/esw.min.js');
        s.onload = function() {
          initESW(null);
        };
        document.body.appendChild(s);
      } else {
        initESW('https://service.force.com');
      }

    </script>

    let isShowOneTimeAllEntered = false
    function allEntered(){
        let isAllEntered = false
        for (let i = 0; i< inputs.length; i++ ){
           if(!inputs[i].value){
               isAllEntered = false
               break;
           }
            isAllEntered = true;
        }
        if(isAllEntered && !isShowOneTimeAllEntered){
            isShowOneTimeAllEntered = true
            customDatalayer('form_allentered')
        }
    }

    let isShowOneTimeEventStart = false
    function eventStart(){
        for(let i = 0; i< inputs.length; i++) {
            if(inputs[i].value && !isShowOneTimeEventStart){
                isShowOneTimeEventStart = true
                customDatalayer('start_form');
            }
        }
    }

        function addFilterTag( id, value ) {
            if (jQuery('#filter-tag-' + id).length) {
                jQuery('#filter-tag-' + id).remove();
            }
            jQuery('.hmg-jb-filter-tags').append('<span id="filter-tag-' + id + '" class="hmg-jb-btn hmg-jb-btn-tag hmg-jb-btn-sm">' + value + ' <button type="button" class="close" aria-label="Clear" onclick="removeFilterTag(\'' + id + '\');resetValue(\'' + id + '\')"><span aria-hidden="true"><i class="jbfa jbfa-times-circle"></i></span></button></span>');
        }

        function removeFilterTag( id ) {
            jQuery('#filter-tag-' + id).remove();
        }

        function resetValue( id ) {

            var is_input = jQuery('input[type="text"]#' + id).length;
            if ( is_input ) {
                jQuery('input[type="text"]#' + id).val('');
            }

            var is_checkbox = jQuery('input[type="checkbox"]#' + id).length;
            if ( is_checkbox ) {
                jQuery('input[type="checkbox"]#' + id).prop("checked", false);
            }

            var is_select = jQuery('option#' + id).length;
            if ( is_select ) {
                jQuery('option#' + id).prop("selected", false);
                jQuery('#category').multipleSelect('refresh');
            }

            var is_combobox = jQuery('#' + id).hasClass('option');

            if ( is_combobox ) {
                jQuery('#' + id).removeClass('selected');
                jQuery('#' + id).attr('aria-selected','false');
                // rebuild the combobox values
                var selecteds = jQuery('#' + id).parent().parent().find(".option.selected");
                var vals = selecteds.map(function() {return jQuery(this).attr('my-data');}).get().join('|');
                jQuery("#" + id).parent().parent().parent().children('input[type="hidden"]').val(vals);
                var combotext = selecteds.map(function() {return jQuery(this).text().trim();}).get().join(', ');
                jQuery("#" + id).parent().parent().parent().children('input[type="text"]').val(combotext);
            }

            loadJBSearchList(1);
        }

        function updatePageFilters() {

            // inputs
            console.log("updatePageFilters");

            jQuery('#JBSearchList_form input[type="text"]').not('.combobox').each(function(index, element) {

                // set IDs
                var elemID = jQuery(element).attr('id');
                if (typeof elemID == typeof undefined || elemID == false) {
                    jQuery(element).attr('id', 'filter-input-' + index);
                    elemID = jQuery(element).attr('id');
                }

                // add tags for inputs
                if (jQuery(element).val()) {
                    addFilterTag(elemID, jQuery(element).val());
                } else {
                    removeFilterTag(elemID);
                }
            });

            // checkboxes (not category)

            jQuery('#JBSearchList_form .hmg-jb-list-group-item input[type="checkbox"]').each(function(index, element) {

                // set IDs
                var elemID = jQuery(element).attr('id');
                if (typeof elemID == typeof undefined || elemID == false) {
                    jQuery(element).attr('id', 'filter-checkbox-' + index);
                    elemID = jQuery(element).attr('id');
                }

                // add tags for checked boxes
                if (jQuery(element).prop( "checked" )) {
                    addFilterTag(elemID, jQuery(element).parent().text());
                } else {
                    removeFilterTag(elemID);
                }
            });

            // select
            var i = 0;
            jQuery('.listbox').find(".option").map(function() {

                console.log("in listbox");


                // set IDs
                var elemID = jQuery(this).attr('id');
                if (typeof elemID == typeof undefined || elemID == false) {
                    jQuery(this).attr('id', 'filter-cat-' + i);
                    elemID = jQuery(this).attr('id');
                }



                // add tags for checked boxes
                if (jQuery(this).hasClass( "selected" )) {
                    console.log("elemID - is selected: " + elemID);
                    addFilterTag(elemID, jQuery(this).text());
                } else {
                    removeFilterTag(elemID);
                }

                i++;

            });

        }

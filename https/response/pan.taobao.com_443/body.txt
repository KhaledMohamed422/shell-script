    $(function () {
        $(window).scroll(function () {
            var wst = $(window).scrollTop();
            for (i = 1; i < 5; i++) {
                if ($("#n" + i).offset().top <= wst + 80) {
                    $('nav li').removeClass("border-warning border-bottom");
                    $(".nav-link").removeClass("text-danger");

                    $("[href='#n" + i + "']").addClass("text-danger");
                    $("[href='#n" + i + "']").parent().addClass("border-warning border-bottom");
                }
            }

        });

        $('.nav-link').click(function () {
            $(this).addClass("text-danger");
            $(this).parent().addClass("border-warning border-bottom");
        });
    });

</script>

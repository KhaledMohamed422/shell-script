var UTM_PARAMS = ['utm_source', 'utm_medium', 'utm_campaign', 'utm_cmpid', 'utm_term', 'utm_content', 'utm_adgroupid', 'gclid'];
var UTM_SOURCE_TO_NETSUITE_CAMPAIGN_ID = {
    'google': '238',
    'facebook': '239',
    'linkedin': '240',
    'bing': '241',
	'email': '242',
};
var NETSUITE_LEAD_SOURCE_WEB_ID = '-6';
 
function saveUTMQueryParams() {
	var queryParams = getQueryParams();
	var utmParams = UTM_PARAMS.reduce(function (allParams, key) {
		if (queryParams[key]) {
        	allParams[key] = queryParams[key];
      	}
      	return allParams;
    	}, {});
/**	if (Object.keys(utmParams).length > 0) {
    	sessionStorage.setItem('utm_params', JSON.stringify(utmParams));
    }*/
}
  
function getQueryParams() {
  	return location.search
    	.replace('?', '')
    	.split('&')
    	.reduce(function(params, pair) {
      		var parts = pair.split('=');
      		var key = parts[0];
      		var value = parts[1];
      		params[key] = decodeURIComponent(value);
      		return params;
    	}, {});
}

function addSessionUtmParamsToAllForms() {
    var fields = {}
    collectSessionUtmParams(fields);
    var currentLocation = window.location;
    fields['hostname'] = currentLocation.hostname
    $("form").map(function (idx, form) {
        for (var fieldName in fields) {
            var fieldValue = fields[fieldName];
            addHiddenFormField(form, fieldName, fieldValue);
        }
        return form.name;
    });
}
  
function collectSessionUtmParams(fields) {
    var utmParamsJSON = sessionStorage.getItem('utm_params');
    if (!utmParamsJSON) {
        return;
    }
    try {
        var utmParams = JSON.parse(utmParamsJSON);
    } catch (error) {
        return;
    }
    for (var key in utmParams) {
        var value = utmParams[key];
        if (value) {
            fields['custentity_' + key] = value;
        }
    }
    var gclid = utmParams['gclid'];
    if (gclid) {
        fields['custentity_utm_gclid'] = gclid;
    }
    var utmSource = utmParams['utm_source'];
    if (utmSource) {
        fields['leadsource'] = (
            UTM_SOURCE_TO_NETSUITE_CAMPAIGN_ID[utmSource] ||
            NETSUITE_LEAD_SOURCE_WEB_ID
        );
    }
}
  
function addHiddenFormField(theForm, key, value) {
    var input = document.createElement('input');
    input.type = 'hidden';
    input.name = key;
    input.value = value;
    theForm.appendChild(input);
}  
</script>

function readCookie(name) {
    var cookiename = name + "=";
    var ca = document.cookie.split(';');
    for(var i=0;i < ca.length;i++) {
        var c = ca[i];
        while (c.charAt(0)==' ') c = c.substring(1,c.length);
        if (c.indexOf(cookiename) == 0) return c.substring(cookiename.length,c.length);
    }
    return null;
}
  
function addMunchkinToAllForms() {
    var cookie = readCookie('_mkto_trk');
    $("form").map(function (idx, form) {
        addHiddenFormField(form, 'munchkinid', cookie);
        return form.name;
    });
}

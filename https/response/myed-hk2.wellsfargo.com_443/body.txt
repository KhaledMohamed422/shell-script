addEventListener("load", function() { setTimeout(hideUrlBar, 0); }, false);

var resizeOffset = 60;



function hideUrlBar() {
    document.body.style.height = (window.innerHeight + resizeOffset) + 'px';
    window.scrollTo(0,1);
}

function hideUrlBarAndResize() {
    hideUrlBar();
    setTimeout(function()
    {
        resize();
    }, 100);
}

function resize() {
    if (document.body.offsetHeight < window.innerHeight) {
        hideUrlBar();
    } else if (document.body.offsetHeight != window.innerHeight || document.body.offsetWidth != window.innerWidth) {
        document.body.style.width = window.innerWidth + 'px';
        document.body.style.height = window.innerHeight + 'px';
        if (document.body.scrollHeight > window.innerHeight) {
            hideUrlBar();
        }
    }

    var footer = document.getElementById('footer');
    var footerHeight = 0;
    if (footer) {
        footerHeight = footer.clientHeight;
    }
    var contentHeight = document.getElementById('main_div').clientHeight;

    if (contentHeight + footerHeight > window.innerHeight) {
        if (footer) {
            footer.style.position = 'static';
            footer.style.top = '';
        }
        document.body.style.height = contentHeight + footerHeight;
    } else if (footer) {
        footer.style.position = 'absolute';
        footer.style.top = (window.innerHeight - footerHeight) + 'px';
    }
}

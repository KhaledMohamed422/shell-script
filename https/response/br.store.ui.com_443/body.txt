    <script>
      const app = new Vue({
        el: '#app',

        i18n,

        data() {
          return {
            appData: window.APP_DATA || {},
            isCartModalOpen: false,
            isCartBannerShown: false,
            breakpoints: {
              MOBILE: 480,
              TABLET: 768,
              LAPTOP: 960,
              DESKTOP: 1200,
            },
            isMobileResolution: false,
            isTabletResolution: false,
            defaultQtyMinimum: 0,
            defaultQtyMaximum: 1000,
            customerProductLimits: null,
            decimalSeparator: ',',
            currentLang: localeCodes[langify.helper.getSelectedLanguage()],
          };
        },

        computed: {
          isDevDomain() {
            return this.appData.devDomains.includes(window.location.host)
          },
        },

        methods: {
          /**
           * @returns {Object[]}
           */
          getBoughtItemQtiesGroupedBySku() {
            const boughtItems = this.appData.boughtItems;
            const boughtItemQties = {};

            for (const { sku, qty } of boughtItems) {
              const oldQty = boughtItemQties[sku] || 0;

              boughtItemQties[sku] = oldQty + qty;
            }

            return boughtItemQties;
          },

          /**
           * Group GET request parameters as an object
           *
           * @returns {Object}
           */
          searchParams() {
            const paramsMap = window.location.search
              .replace('?', '')
              .split('&')
              .map(param => param.split('='));

            // Convert the params map to an object
            const params = {};

            for (const [key, value] of paramsMap) {
              params[key] = value;
            }

            return params;
          },

          /**
           * Set resolution booleans
           */
          checkResolutions() {
            const ww = window.outerWidth;

            this.isMobileResolution = ww <= this.breakpoints.MOBILE;
            this.isTabletResolution = ww <= this.breakpoints.TABLET;
          },

          /**
           * Helper method to fetch asset with typecheck
           */
          asset(key, fallback = null) {
            const asset = this.appData.assets[key];

            return typeof asset !== 'undefined' ? asset : fallback;
          },

          /**
           * Open the cart modal
           */
          openCart() {
            this.isCartModalOpen = true;
          },

          /**
           * Close the cart modal
           */
          closeCart() {
            this.isCartModalOpen = false;
          },

          /**
           * Show the cart banner
           */
          showCartBanner() {
            this.isCartBannerShown = true;
          },

          /**
           * Hide the cart banner
           */
          hideCartBanner() {
            this.isCartBannerShown = false;
          },

          /**
           * Helper method to format money
           *
           * @param {number} cents
           * @param {applyVat} boolean
           * @returns {string}
           */
          formatMoney(cents, applyVat = false) {
            // Format money
            const money = Shopify.formatMoney(cents, false, applyVat)
            const parts = money.split(' ')
            const moneyPart = parts[1].split(
              this.decimalSeparator,
            )

            const primary = moneyPart[0]
            const decimal = moneyPart[1] || ''
            const prefix = parts[0] || ''


            // Combine parts to render
            const html = []

            if (prefix) {
              html.push(`<span>${prefix}</span>&nbsp;`)
            }

            html.push(`<span>${primary}</span>`)

            if (decimal !== '00') {
              html.push(`<span class="smaller">${this.decimalSeparator}${decimal}</span>`)
            }

            return html.join('')
          },

          formatMoneySimple(cents, applyVat = false) {
            // Format money
            const money = Shopify.formatMoney(cents, false, applyVat)
            const parts = money.split(' ')
            const moneyPart = parts[1].split(
              this.decimalSeparator,
            )

            const primary = moneyPart[0]
            const decimal = moneyPart[1] || ''
            const prefix = parts[0] || ''

            // Combine parts to render
            const html = []

            if (prefix) {
              html.push(`${prefix} `)
            }

            html.push(`${primary}`)

            if (decimal !== '00') {
              html.push(`${this.decimalSeparator}${decimal}`)
            }

            return html.join('')
          },

          /**
           * Set-up all cart related actions
           */
          initCart() {
            // Initiate CartJS
            CartJS.init(this.appData.cart, {
              'requestBodyClass': 'loading',
              'moneyFormat': 'R$ {{amount_no_decimals}}',
              'moneyWithCurrencyFormat': 'R$ {{amount_no_decimals}} BRL'
            });

            // Open the cart if requested
            if (window.location.search === "?cart") {
              window.history.replaceState({}, null, window.location.pathname);
              this.openCart();
            }

            // Handle cart header menu item
            jQuery(document).on('click', '.site-header__cart-toggle', e => {
              e.preventDefault();
              this.openCart();
            });
          },

          setProductSku() {
            const { product } = this.appData;
            const { variant: variantId } = this.searchParams();
            const skuEl = document.querySelector('.product-current-sku');

            if (product && skuEl) {
              // Use the first variant by default
              let variant = product.variants[0];

              if (variantId) {
                // Find the variant based on the request parameter
                variant = product.variants.find(
                  variant => parseInt(variant.id) === parseInt(variantId)
                );
              }

              if (variant) {
                // Update the SKU
                let gen2 = new RegExp('-gen2', 'i');
                skuEl.innerText = variant.sku.replace(gen2, '');
              }
            }
          },

          makeSlug(text) {
            return text.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/-$/, '').replace(/^-/, '');
          },

          getUrlParameter(sParam) {
            const sPageURL = decodeURIComponent(window.location.search.substring(1))
            const sURLVariables = sPageURL.split('&')

            for (const variable of sURLVariables) {
              const [key, value] = variable.split('=')

              if (key === sParam) {
                return typeof value === 'undefined' ? true : value
              }
            }
          },

          replaceUrlParam(e,r,a){
            var n=new RegExp("("+r+"=).*?(&|$)"),c=e;return c=e.search(n)>=0?e.replace(n,"$1"+a+"$2"):c+(c.indexOf("?")>0?"&":"?")+r+"="+a
          },

          /**
           * Handle [data-toggle] that toggles [data-toggle-id] elements
           */
          handleDataToggle(e) {
            const $action = jQuery(e.currentTarget);
            const targetId = $action.data('toggle');
            const $target = jQuery(`[data-toggle-id="${targetId}"]`);

            const toggleText = $action.attr('data-toggle-text');
            const toggleTextNew = $action.attr('data-toggle-text-new');
            const toggleTextTarget = $action.children('[data-toggle-text-target]').first();

            const forceOpen = typeof $action.data('toggle-open') !== 'undefined';
            const forceClose = typeof $action.data('toggle-close') !== 'undefined' && !forceOpen;

            e.preventDefault();

            if ($action.hasClass('active') || forceClose) {
              if (!forceOpen) {
                $target.removeClass('active');
                $target.stop(true).slideUp(300);
                $action.removeClass('active');
              }
            } else if (!$action.hasClass('active') || forceOpen) {
              if (!forceClose) {
                $target.addClass('active');
                $target.stop(true).slideDown(300);
                $action.addClass('active');
              }
            }

            if(!!toggleText && !!toggleTextNew && !!toggleTextTarget) {
              $action.hasClass('active') ? toggleTextTarget.text(toggleTextNew) : toggleTextTarget.text(toggleText);
            }

          },

          /**
           * Handle [data-scroll-to] that scrolls to [data-scroll-to-id] elements
           */
          handleDataScrollTo(e) {
            const $action = jQuery(e.currentTarget);
            const targetId = $action.data('scroll-to');
            const $target = jQuery(`[data-scroll-to-id="${targetId}"]`);

            jQuery('html, body').animate({
              scrollTop: $target.offset().top,
            }, 600);
          },

          /**
           * Fetch and parse a cart attribute
           *
           * @param {string} name
           * @returns {Object}
           */
          getCartAttribute(name) {
            const attributes = CartJS.getAttributes();
            const attribute = attributes && attributes[name] ? attributes[name] : '{}';

            return JSON.parse(attribute.replace(/=>/g, ':'));
          },

          /**
           * @param {boolean} forceUpdate
           * @returns {Promise<{ sku: string; max: string }[]>}
           */
          async getCustomerProductLimits(forceUpdate = false) {
            if (!this.customerProductLimits || forceUpdate) {
              const response = await fetch('/apps/shop-manager/api/product-limits', {
                method: 'POST',
                headers: {
                  Accept: 'application/json',
                  'Content-Type': 'application/json',
                },
                body: JSON.stringify({ shop: Shopify.shop }),
              })

              try {
                this.customerProductLimits = await response.json();
              } catch (error) {
                console.error(error)
                this.customerProductLimits = [];
              }

            }

            return this.customerProductLimits;
          },

          /**
           * Handle login referral redirect
           *
           * @returns {void}
           */
          handleLoginReferral() {
            if (!localStorage) {
              return
            }

            // If referral exists and not on login page
            const accountLoginRef = localStorage.getItem('account-login-ref')
            const isLoginPage = location.pathname === '/account/login'

            if (!accountLoginRef || isLoginPage) {
              return
            }

            // Remove the item
            localStorage.removeItem('account-login-ref')

            // Only if on account page (after login)
            const isAccountPage = location.pathname === '/account'

            if (!isAccountPage) {
              return
            }

            // Redirect to the referred url
            window.location = accountLoginRef
          },

          redirectToLogout() {
            if (this.isDevDomain) {
              window.location = '/account/logout'
            } else {
              window.location = `https://sso.ui.com/api/sso/v1/logout?redirect=https%3A%2F%2F${window.location.host}%2Faccount%2Flogout`
            }
          },

          redirectToLogin() {
            // Set a local storage referral url to current page
            if (localStorage) {
              localStorage.setItem('account-login-ref', location.href)
            }
            // Redirect
            location.href = '/account/login';
          },
        },

        created() {
          this.checkResolutions();
          this.initCart();
        },

        mounted() {
          this.setProductSku();
          this.handleLoginReferral()

          // Events
          jQuery(() => {
            // Show cart banner when document is loaded
            this.isCartBannerShown = true
          })
          jQuery(window).on('resize', this.checkResolutions);
          jQuery(document).on('click', '[data-toggle]', this.handleDataToggle);
          jQuery(document).on('click', '[data-scroll-to]', this.handleDataScrollTo);

          // pushState event hack
          const pushState = history.pushState;
          history.pushState = (...args) => {
            pushState.apply(history, args);

            // Custom events
            this.setProductSku();
          };
        },
      });
    </script>

  computed: {
    hasCurrencySelect() {
      const shopOrigins = ['store-ui-europe.myshopify.com']

      return this.shop !== null && shopOrigins.includes(this.shop.origin)
    },

    currencySelectCurrencies() {
      const result = []

      if (this.hasCurrencySelect) {
        for (const { currency } of this.shop.countries) {
          if (!result.includes(currency)) {
            result.push(currency)
          }
        }
      }

      return result
    },
  },

  methods: {
    hasCookie() {
      return typeof jQuery.cookie(this.cookieKey) !== 'undefined'
    },

    setCookie(closeAction = false) {
      jQuery.cookie(this.cookieKey, closeAction ? '0' : '1', {
        expires: 30,
        path: '/',
      })
    },

    close() {
      // Set the cookie
      this.setCookie()

      // Hide the modal
      this.canShow = false
    },

    openShop() {
      if (!this.shop) {
        return
      }

      // Set the cookie
      this.setCookie()

      // Redirect
      const shopUrlParams = this.hasCurrencySelect
        ? '?' +
          [
            `refCountry=${this.selectedCountryCode}`,
            `refCurrency=${this.selectedCurrencyCode}`,
          ].join('&')
        : ''

      window.location = `${this.shop.url}${shopUrlParams}`
    },

    async initGeoCountry() {
      if (this.geoCountry) {
        return
      }

      try {
        const url = 'https://www.ui.com/api/geoip'
        const res = await fetch(url, { method: 'POST' })
        const { country_code: code, country_name: name } = await res.json()

        // Set the country as the selected country
        this.selectedCountryCode = code
        this.geoCountry = { code, name }
      } catch (e) {
          console.error('Failed to get country. ', e)
      }
    },

    initShop() {
      if (this.shop || !this.geoCountry) {
        return
      }

      for (const shop of this.shops) {
        const hasCountry = shop.countries.some(
          ({ code }) => code === this.geoCountry.code,
        )

        if (hasCountry) {
          this.shop = shop
          break
        }
      }
    },

    handleRefParams() {
      const { origin, searchParams } = new URL(location.href)
      let reloadPage = false

      // Set country
      const refCountry = searchParams.get('refCountry')

      if (refCountry) {
        jQuery.cookie('site-country', refCountry, {
          path: '/',
        })
        reloadPage = true
      }

      // Set currency
      const refCurrency = searchParams.get('refCurrency')

      if (refCurrency) {
        jQuery.cookie('site-currency', refCurrency, {
          path: '/',
        })
        reloadPage = true
      }

      // Reload the page to update content and remove parameters
      if (reloadPage) {
        window.location = origin
      }
    },
  },

    // Init data
    await this.initGeoCountry()
    this.initShop(this.geoCountry)

      // Handle ref parameters

<script>
  const carouselApp = {
    state: {
      // Index 1
      currentSlide: 1,
      autoRotateInterval: null,
      autoRotateDirection: 'right',

      animationTimeout: null,
      isAnimating: false,
      touchstartX: 0,
      touchendX: 0,
    },

    elements: {
      container: document.getElementById('bannerCarousel_slidesContainer'),
      slides: document.getElementById('bannerCarousel_slides'),
      nav: document.getElementById('bannerCarousel_nav'),
      navRight: document.getElementById('bannerCarousel_navRight'),
      navLeft: document.getElementById('bannerCarousel_navLeft'),
      footer: document.getElementById('bannerCarousel_footer'),
    },

    // Get count of slides in carousel
    getSlidesCount() {
      return this.elements.slides.childElementCount
    },

    // Get width of the overflow container in carousel
    getContainerWidth() {
      return this.elements.container.getBoundingClientRect().width
    },

    handleGesture() {
      const { touchendX, touchstartX } = this.state
      const touchSlideDistance = (touchstartX - touchendX)

      if (touchSlideDistance < -50) {
        this.slideLeft(true)
      } else if (touchSlideDistance > 50) {
        this.slideRight(true)
      } else {
        this.slideTo(this.state.currentSlide, true)
      }
    },

    // Helper to slide one step left
    slideLeft(cancelInterval = false) {
      const { currentSlide } = this.state
      const { slides } = this.elements

      if (currentSlide > 1) {
        this.slideTo(currentSlide - 1, cancelInterval)
      } else {
        this.slideTo(currentSlide)
      }
    },

    // Helper to slide one step right
    slideRight(cancelInterval = false) {
      const { currentSlide } = this.state
      const { slides } = this.elements

      if (currentSlide < this.getSlidesCount()) {
        this.slideTo(currentSlide + 1, cancelInterval)
      } else {
        this.slideTo(currentSlide)
      }
    },

    // Slide to slide number - (index 1)
    slideTo(slideNum, cancelInterval = false) {
      clearInterval(this.state.animationTimeout)
      this.state.isAnimating = true

      this.state.animationTimeout = setTimeout(() => {
        this.state.isAnimating = false
      }, 800)

      if (cancelInterval) {
        clearInterval(this.state.autoRotateInterval)
      }

      const { navLeft, navRight } = this.elements

      if (slideNum === 1) {
        navLeft.classList.add('disabled')
      } else {
        navLeft.classList.remove('disabled')
      }

      if (slideNum === this.getSlidesCount()) {
        navRight.classList.add('disabled')
      } else {
        navRight.classList.remove('disabled')
      }

      this.state.currentSlide = slideNum

      const leftPos = this.getContainerWidth() * (slideNum - 1)
      this.elements.slides.style.left = `${-leftPos}px`

      this.updateActiveThumb()
    },

    // Set the static width of slides
    setSlidesWidth() {
      const { slides, container } = this.elements
      const containerWidth = this.getContainerWidth()

      this.slideTo(1)

      slides.style.width = `${containerWidth}px`
    },

    // Update thumb elements to show current active slide
    updateActiveThumb() {
      const { currentSlide } = this.state
      const { footer } = this.elements

      let i = 0
      for (const el of footer.children) {
        if (i + 1 === currentSlide) {
          el.classList.add('active')
        } else {
          el.classList.remove('active')
        }

        i += 1
      }
    },

    initSlideThumbs() {
      const { footer } = this.elements
      const indicatorHTML = `<svg width="9" height="8" viewBox="0 0 9 8" fill="none" xmlns="http://www.w3.org/2000/svg"><circle cx="4.5" cy="4" r="4" fill="#C8CAD3"/></svg>`

      for (let i = 0; i < this.getSlidesCount(); i++) {
        const el = document.createElement('button')

        el.innerHTML = indicatorHTML
        el.id = `bannerCarouselThumb_${i + 1}`
        el.classList.add('banner-carousel__thumb')

        el.addEventListener('click', () => {
          this.slideTo(i + 1, true)
        })

        footer.appendChild(el)
      }

      this.updateActiveThumb()
    },

    initInterval() {
      this.state.autoRotateInterval = setInterval(() => {
        const { currentSlide, autoRotateDirection } = this.state
        const slidesCount = this.getSlidesCount()

        if (autoRotateDirection === 'right' && currentSlide < slidesCount) {
          this.slideRight()
        } else if (autoRotateDirection === 'right') {
          this.state.autoRotateDirection = 'left'
          this.slideLeft()
        } else if (autoRotateDirection === 'left' && currentSlide > 1) {
          this.slideLeft()
        } else if (autoRotateDirection === 'left') {
          this.state.autoRotateDirection = 'right'
          this.slideRight()
        }
      }, 6000)
    },

    init() {
      const {
        container,
        slides,
        navRight,
        navLeft,
        nav,
      } = this.elements

      this.setSlidesWidth()
      this.slideTo(1)
      this.initInterval()

      if (this.getSlidesCount() > 1) {
        this.initSlideThumbs()
      }

      window.addEventListener('resize', this.setSlidesWidth.bind(this))
      navRight.addEventListener('click', () => this.slideRight.bind(this)(true))
      navLeft.addEventListener('click', () => this.slideLeft.bind(this)(true))

      /**
       * Record touchstart X positon
       */
      container.addEventListener('touchstart', e => {
        if (this.state.isAnimating) {
          this.state.touchstartX = null
          return
        }

        slides.style.transition = 'none'
        this.state.touchstartX = e.changedTouches[0].screenX
      })

      /**
       * Visual "drag" animation
       */
      container.addEventListener('touchmove', e => {
        if (typeof this.state.touchstartX !== 'number') {
          return
        }

        const draggedDistance = e.changedTouches[0].screenX - this.state.touchstartX
        const currentSlideLeftPos = 0 - (this.getContainerWidth() * (this.state.currentSlide - 1))

        const draggedLeftPos = currentSlideLeftPos + (draggedDistance / 2)
        const minLeftPos = 0 - (this.getContainerWidth() * (this.getSlidesCount() - 1))

        if (draggedLeftPos < 0 && draggedLeftPos > minLeftPos) {
          slides.style.left = `${draggedLeftPos}px`
        }
      })

      /**
       * Record touchend X positon and check if valid gesture for carousel slide
       */
      container.addEventListener('touchend', e => {
        if (typeof this.state.touchstartX !== 'number') {
          return
        }

        slides.style.transition = 'left .8s'
        this.state.touchendX = e.changedTouches[0].screenX
        this.handleGesture()
      })

      nav.style.display = 'block'
    },
  }

  window.addEventListener('load', () => {
    carouselApp.init()
  })
</script>

/**
 * 通用的前端页面跳转拦截处理模块，对 location.href, open 等跳转进行拦截处理
 * @param {*} handleHref 处理href的钩子，(href, options) => href，入参href已自动转绝对路径
 */
window.interceptNavigation = function(handleHref) {
    /**
     * 底层跳转拦截钩子，统一对跳转URL进行处理
     * @param {string} href 要跳转的地址，支持相对路径、绝对路径
     * @param {boolean} options {type, a?} type定义包含：a, location, open, pushState, replaceState，a表示A标签跳转时的DOM对象
     * @return {string} 返回处理好的跳转地址，如果没有返回值则表示不需要继续处理跳转
     */
    function beforeRedirect(href, options) {
        if (!href) {
            return;
        }
        // 传过来的href可能是相对路径，为了做统一判断需要转成绝对路径
        if (href.indexOf('http') !== 0) {
            var a = document.createElement('a');
            a.href = href;
            href = a.href;
        }
        if (handleHref) {
            href = handleHref(href, options);
        }
        return href;
    }

    // A标签跳转拦截
    document.addEventListener('click', function (e) {
        var target = e.target || {};
        // A标签可能包含子元素，点击目标可能不是A标签本身，这里只简单判断2层
        if (target.tagName === 'A' || (target.parentNode && target.parentNode.tagName === 'A')) {
            target = target.tagName === 'A' ? target : target.parentNode;
            var href = target.href;
            // 不处理没有配置href或者指向JS代码的A标签
            if (!href || href.indexOf('javascript') === 0) {
                return;
            }
            var newHref = beforeRedirect(href, { type: 'a', a: target });
            // 没有返回值一般是已经处理了跳转，需要禁用当前A标签的跳转
            if (!newHref) {
                target.target = '_self';
                target.href = 'javascript:;';
            } else if (newHref !== href) {
                target.href = newHref;
            }
        }
    }, true);

    // location.href 拦截至今是一个困扰前端界的难题，这里只能采用一个折中的方法
    // 由于 location.href 无法重写，只能实现一个 location2.href = ''
    // 建议代码中所有使用到了 location.href 的地方都替换成 (window.location2 || window.location).href
    if (Object.defineProperty) {
        window.location2 = {};
        Object.defineProperty(window.location2, 'href', {
            get: function() {
                return location.href;
            },
            set: function(href) {
                var newHref = beforeRedirect(href, { type: 'location' });
                if (newHref) {
                    location.href = newHref;
                }
            },
        });
    }

    // window.open 拦截
    var tempOpenName = '_rawOpen';
    if (!window[tempOpenName]) {
        window[tempOpenName] = window.open;
        window.open = function(url, name, features) {
            url = beforeRedirect(url, { type: 'open' });
            if (url) {
                window[tempOpenName](url, name, features);
            }
        };
    }

    // history.pushState 拦截
    var tempPushName = '_rawPushState';
    if (!window.history[tempPushName]) {
        window.history[tempPushName] = window.history.pushState;
        window.history.pushState = function(state, title, url) {
            url = beforeRedirect(url, { type: 'pushState' });
            if (url) {
                window.history[tempPushName](state, title, url);
            }
        };
    }

    // history.replaceState 拦截
    var tempReplaceName = '_rawReplaceState';
    if (!window.history[tempReplaceName]) {
        window.history[tempReplaceName] = window.history.replaceState;
        window.history.replaceState = function(state, title, url) {
            url = beforeRedirect(url, { type: 'replaceState' });
            if (url) {
                window.history[tempReplaceName](state, title, url);
            }
        };
    }
};

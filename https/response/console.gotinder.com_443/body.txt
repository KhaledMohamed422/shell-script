        var el = document.createElement(type === 'css' ? 'link' : 'script');
        el.type = type === 'css' ? 'text/css' : 'text/javascript';
        if (type === 'css') {
            el.rel = 'stylesheet';
            el.href = publicPath + filename;
        } else {
            el.src = publicPath + filename;

            };

    }

    function isCdnCheckRequired(){
        var result = true;
        var week = 7*24*3600*1000;
        
        try {
            var item = JSON.parse(localStorage.getItem('ciqCdnBlocked'));
            
            result = item.cdnUrl === window.webpackManifest.publicPath &&
                Date.now() - item.timestamp < week &&
                item.available !== true;

    }

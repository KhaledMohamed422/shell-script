            const boughtQty = this.boughtItemsQties[limit.sku] || 0

            if (limit.allowOne === 'yes' && boughtQty >= limit.max) {
              tmpLimits[i].max = boughtQty + 1
            }
          }
        }

        this.customerProductLimits = tmpLimits
        return tmpLimits
      },

      /*
       * Helper FN
       */
      combineLimits(generalLimits = [], customerLimits = []) {
        const combined = generalLimits.map(generalLimit => {
          const found = customerLimits.find(customerLimit => customerLimit.variantId === generalLimit.variantId)

          if (found) {
            const combinedMax = parseInt(generalLimit.max) + parseInt(found.max)
            return { ...generalLimit, max: combinedMax.toString() }
          }

          return generalLimit
        })

        const customerLimitsNotAppended = customerLimits.filter(customerLimit =>
          !combined.some(x => x.variantId === customerLimit.variantId)
        )

        const mappedCustomerLimits = customerLimitsNotAppended.map(x => {
          return {
            max: x.max,
            sku: x.sku,
            variantId: x.variantId,
          }
        })

        return [...combined, ...mappedCustomerLimits]
      },

      /*
       * Check if variant has a set limit
       */
      async isVariantLimitEnabled(id, sku) {
        const customerProductLimits = await this.getCustomerProductLimits()

        if (Array.isArray(customerProductLimits)) {
          return customerProductLimits.some(item =>
            item.sku === sku || `${item.variantId}` === `${id}`
          )
        }

        return false
      },

      getQuantityInCart(sku) {
        let cartItem

        if (this.cart) {
          if(Array.isArray(sku)) {
            cartItems = this.cart.items.filter(item => sku.includes(item.sku)) || []
            cartItem = cartItems.reduce((prev, curr) => prev.quantity < curr.quantity ? prev : curr, 0)
          } else {
            cartItem = this.cart.items.find(item => item.sku === sku)
          }
        }

        return cartItem ? cartItem.quantity : 0
      },

      /*
       * Returns the original max limit for variant
       */
      async getVariantCustomerMax(id, sku) {
        const customerProductLimits = await this.getCustomerProductLimits()

        if (Array.isArray(customerProductLimits)) {
          const customerVariantLimit = customerProductLimits.find(
            (item) => item.sku === sku || `${item.variantId}` === `${id}`,
          )

          if (customerVariantLimit) {
            return parseInt(customerVariantLimit.max)
          }
        }

        return this.defaultQtyMaximum
      },

      /*
       * Returns the current max limit for variant
       * (Limit - Bought)
       */
      async getVariantCustomerLimit(sku) {
        // Unknown ID
        const id = 0

        const limitsEnabled = await this.isVariantLimitEnabled(id, sku)
        const variantCustomerMax = await this.getVariantCustomerMax(id, sku)

        if (limitsEnabled) {
          if (!this.isLoggedIn) {
            return Math.min(this.defaultQtyMaximum, variantCustomerMax)
          }

          const boughtQty = this.boughtItemsQties[sku] || 0
          const diff = variantCustomerMax - boughtQty

          if (diff < 0) {
            return 0
          }

          return diff
        }

        return variantCustomerMax
      },

      async setLimits() {
        const variant = this.currentVariant || this.variants && this.variants[0] || this
        console.log('->',variant)
        if (variant) {
          this.maxPerCustomer = await this.getVariantCustomerMax(variant.id, variant.sku)
          this.maxPerCustomerLimit = await this.getVariantCustomerLimit(variant.sku)
          this.variantHasCustomerLimit = await this.isVariantLimitEnabled(variant.id, variant.sku)
        }
      },
    },
  }

          getBoughtItemQtiesGroupedBySku() {
            const boughtItems = this.appData.boughtItems;
            const boughtItemQties = {};

            for (const { sku, qty } of boughtItems) {
              const oldQty = boughtItemQties[sku] || 0;

              boughtItemQties[sku] = oldQty + qty;
            }

            return boughtItemQties;

          /**
           * Group GET request parameters as an object

           * @returns {Object}
           */
          searchParams() {
            const paramsMap = window.location.search
              .replace('?', '')
              .split('&')
              .map(param => param.split('='));

            // Convert the params map to an object
            const params = {};

            for (const [key, value] of paramsMap) {
              params[key] = value;
            }

            return params;
          },

          /**
           * Set resolution booleans
           */
          checkResolutions() {
            const ww = window.outerWidth;

            const moneyPart = parts[0].split(
              this.decimalSeparator,
            )

            const primary = moneyPart[0]
            const decimal = moneyPart[1] || ''
            const suffix = parts[1] || ''

            // Combine parts to render

            if (decimal) {
              html.push(`<span class="smaller">${this.decimalSeparator}${decimal}</span>`)
            }
            if (suffix) {
              html.push(`&nbsp;<span>${suffix}</span>`)
            }

            return html.join('')
          },

    * function updateClock()
    * Update the time display

    function updateClock() {
        var currentTime, currentHours, currentMinutes, currentSeconds, newDate,
            timeoutMillis = 1000, timeoutMaxInterval = 30;

        newDate = new Date().getTime();
        currentTime = new thirdparty.Date();

       //timezone for Central Europe Time from wich "Europe/Copenhagen" is part of it
        currentTime.setTimezone(betfair2olsonMapping["CET"]);
        currentTime.setTime(newDate);

        currentHours = currentTime.getHours();
        currentMinutes = currentTime.getMinutes();
        currentSeconds = currentTime.getSeconds();

        currentMinutes = ( currentMinutes < 10 ? "0" : "" ) + currentMinutes;
        currentHours = ( currentHours < 10 ? "0" : "" ) + currentHours;
        currentSeconds = ( currentSeconds < 10 ? "0" : "" ) + currentSeconds;

        // set the timeout interval to the timeoutMaxInterval value or the remaining number of seconds from now to the timeoutMaxInterval value
        timeoutMillis = timeoutMillis * (currentSeconds < timeoutMaxInterval ? timeoutMaxInterval - currentSeconds : 60 - currentSeconds);

        footerClockTag.innerHTML = currentHours + ":" + currentMinutes;

        setTimeout(updateClock, timeoutMillis);
    }

            if (footerClockTag) {
                betfair.cssjs("add", footerClockTag, "enabled");

                updateClock();
            }
        }
    }

betex.util.Analytics.LoginObserver.prototype.getParamsToStore = function (sourceParams) {
	// clone sourceParams so we leave it untouched

betex.util.Analytics.LoginObserver.prototype.mergeObjects = function (masterObj, childObj) {
	//child object's properties will supersede any shared properties of the master
	var property, mergeObj = {};

	for (property in masterObj) {
		if (masterObj.hasOwnProperty(property)) {
			mergeObj[property] = masterObj[property];
		}
	}

	for (property in childObj) {
		if (childObj.hasOwnProperty(property)) {
			mergeObj[property] = childObj[property];
		}
	}
	return mergeObj;

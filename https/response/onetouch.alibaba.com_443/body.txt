    (function () {
        var header = document.querySelector(".homepage-global-header");
        var nav = header.querySelector(".nav");
        var category,// 当前频道站
            mode;// 导航状态默认沉浸式  Immersive 沉浸式   Normal 正常

        if (window.__header__) {
            category = window.__header__.category || "home";
            mode = window.__header__.mode || "Immersive";
        }
        
        if (window.__global_header_mode__ === "normal") {
            mode = "Normal";
        }

        function getScrollTop() {
            return document.documentElement.scrollTop || window.pageYOffset || document.body.scrollTop;
        }

        function handlePageHeader() {
            var categoryList = nav.querySelectorAll(".nav-list>li[class='nav-" + category + "']");

            if (category !== "home") {
                document.querySelector(".homepage-global-header .bar-back").style.visibility = "visible";
            } else {
                nav.querySelector("li.nav-more").style.display = "block"; 
            }

            for (let i = 0; i < categoryList.length; i++) {
                var item = categoryList[i];
                item.style.display = "block"; 
            }

            nav.style.display = "block";
            
        }

        function handleUserInfo() {
            var login = document.querySelector(".homepage-global-header .bar-login");
            var singout = document.querySelector(".homepage-global-header .bar-singout");

            if (window.__cookie_info__ && window.__cookie_info__.isLoggedIn) {
                login.style.display = "none";
                singout.querySelector(".bar-username span").innerText = window.__cookie_info__.firstName || 'NUll';
                singout.style.display = "block";
            }
        }

        function handleNavIndex() {
            var navLi = nav.querySelectorAll(".nav-list > li");

            for (let i = 0; i < navLi.length; i++) {
                var item = navLi[i];
                if (window.getComputedStyle(item).display !== "none") {
                    var childList = item.querySelectorAll(".nav-item");

                    for (let j = 0; j < childList.length; j++) {
                        var child = childList[j];
                        var path = child.dataset['urlPath'];
                        var link = location.hostname + location.pathname;
                        if (path && link === path) {
                            item.className += " active";
                        }
                    }
                }
            }
        }

        var isTop = getScrollTop() === 0;

        function handleScroll() {
            if (mode === "Normal" || (getScrollTop() > 0 && isTop)) {
                isTop = false;
                header.className += ' Normal';
            } else if (getScrollTop() <= 0 && !isTop) {
                isTop = true;
                header.className = header.className.replace(/ Normal/g, "");
            }
        }
        
        function handleNavCommon() {
            var commonList = nav.querySelectorAll("li.nav-common");
            for (let i = 0; i < commonList.length; i++) {
                var child = commonList[i];
                var path = child.querySelector(".nav-item").dataset['urlPath'];
                
                if(category === "home" && path === 'onetouch.alibaba.com/home/live') {
                    child.style.display = "none";
                } else {
                    child.style.display = "block";
                }
            }
        }


        handlePageHeader();
        handleNavIndex();
        handleNavCommon();
        handleScroll();

        window.onload = function () {
            handleUserInfo();
        }

        if (mode === "Immersive") {
            window.onscroll = function () {
                handleScroll();
            }
        }
    })();

	<script type="text/javascript" nonce="">
		// Hide the redirection notice (to prevent a flash)
		document.getElementById('notice').style.display = 'none';

		var url = window.location;
		var whereTo = '';
		var query = '';
		var redirectMap = [{
			pattern: /^\/people\/contacts\/?$/,
			dest: '/people/me/contacts/'
		}, {
			pattern: /^\/people(?:\/page\d+)?\/?$/,
			dest: '/'
		}, {
			pattern: /^\/home\/?$/,
			dest: '/'
		}, {
			pattern: /^\/explore\/interesting(?:\/page\d+)?\/?$/,
			dest: '/explore/'
		}, {
			pattern: /^\/search\/advanced.*\/?$/,
			dest: '/search/'
		}, {
			pattern: /^\/photos\/[^\/]+\/more\/?$/,
			dest: '/photos/me/'
		}];

		// Donâ€™t ever jump to any URL with an http in it
		function validatePath(str) {
			return str && str.indexOf('http') === -1;
		}

		// Prefer the #hash, but fall back to the path.
		if (validatePath(url.hash)) {
			whereTo = url.hash.replace('#', '');
		} else if (validatePath(url.pathname)) {
			whereTo = url.pathname;
			query = url.search;
		}

		// Hunt down query params
		if (!query) {
			// They might be in _QM_q_IS_dogs_AND_ss_IS_0_AND_mt_IS_all_AND_w_IS_all format inside the path
			whereTo = whereTo.replace(/_QM_/gi, '?').replace(/_AND_/gi, '&').replace(/_IS_/gi, '=').split('?');

			if (whereTo[1]) {
				query = '?' + whereTo[1];
			}

			whereTo = whereTo[0];
		}

		// Ensure a well formed path
		if (whereTo && whereTo[0] !== '/') {
			whereTo = '/' + whereTo;
		}

		// Check to see if this URL is rewritten
		for (var i in redirectMap) {
			if (redirectMap[i].pattern.test(whereTo)) {
				if (redirectMap[i].dest) {
					whereTo = redirectMap[i].dest;
				}

				break;
			}
		}

		// Reappend query params
		whereTo += query;

		// Go go go!
		location = 'https://www.flickr.com' + whereTo;
	</script>

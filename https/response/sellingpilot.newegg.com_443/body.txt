  <script>
    var authed = false, authedEmail = '';
    function signin() { authed ? location.href = '/account' : euf.user.login() }
    function toggleFeedbackDialog(display, tool) { window.dispatchEvent(new CustomEvent('sp:togglefeedbackdialog', { detail: { display: display === true, tool }, })) }

    function toggleDropdown() {
      var el = document.querySelector('div.screen.small>.dropdown-menus');
      if (!el) return;
      el.classList.contains('show') ? el.classList.remove('show') : el.classList.add('show');
    }

    (function () {
      window.addEventListener('scroll', function (evt) {
        var scrollTop = document.documentElement.scrollTop || document.body.scrollTop;
        if (Number.isNaN(scrollTop)) return
        document.body.classList[scrollTop > 100 ? 'add' : 'remove']('scrolled');
      });

      euf.user.getUserInfo().then(function (data) {
        authed = !!data;
        if (authed) {
          var nodes = document.querySelectorAll('*[signinbtn]');
          for (var i = 0; i < nodes.length; i++) { nodes[i].innerText = data.LoginName }
        }
        authedEmail = authed ? data.LoginName || '' : '';
      }).catch(function () { })

      var langSelector = document.getElementById('lang-selector');
      euf.i18n.onLanguageCodeChange(function (lang) {
        if (langSelector) langSelector.setAttribute('data-langtext', lang);
      });

      var currentLang = '';
      if (location.pathname === '/') {
        currentLang = 'en-us';
      } else {
        switch (location.pathname.toLowerCase()) {
          case '/en-us':
            currentLang = 'en-us';
            break;
          case '/zh-cn':
            currentLang = 'zh-cn';
            break;
          case '/zh-tw':
            currentLang = 'zh-tw';
            break;
        }
      }
      if (['en-us', 'zh-cn', 'zh-tw'].indexOf(currentLang) === -1) currentLang = 'en-us';
      euf.i18n.setLanguageCode(currentLang);

      new Swiper('.swiper', { autoplay: { delay: 3000 }, loop: true, pagination: { el: '.swiper-pagination' } });
    })();
  </script>

  <script>
    (function () {
      const app = Vue.createApp({
        data() {
          return { showFeedback: false, formSubmitted: false, feedback: { email: '', suggestion: '', tool: '' }, feedbackSubmitted: false }
        },
        created() {
          var self = this;
          window.addEventListener('sp:togglefeedbackdialog', function (evt) {
            if (!evt.detail || !evt.detail.tool) return;
            evt.detail.display === true ? self.showFeedbackDialog(evt.detail.tool) : self.hideFeedbackDialog();
          });
        },
        computed: {
          errors() {
            var result = { Email: '', Suggestion: '' };
            if (!this.feedback.email) {
              result.Email = 'Please fill in your email address';
            } else if (!/^[\w-]+(\.[\w-]+)*@[a-zA-Z0-9]+(\.[a-zA-Z0-9]+)*(\.[a-zA-Z]{2,})$/.test(this.feedback.email)) {
              result.Email = 'Invalid email address';
            }
            return result;
          },
          invalid() { return Object.values(this.errors).join('').length > 0 }
        },
        methods: {
          saveFeedback() {
            this.formSubmitted = true;
            if (this.invalid) return;
            var self = this;

            euf.globalLoading.show();
            var xhr = new XMLHttpRequest();
            xhr.open('POST', '/api/feedback/follow');
            xhr.onreadystatechange = function (ev) {
              if (xhr.readyState !== XMLHttpRequest.DONE) return;
              euf.globalLoading.hide();
              self.feedbackSubmitted = xhr.status <= 300;
              self.hideFeedbackDialog();
            };
            xhr.setRequestHeader('content-type', 'application/json');
            xhr.send(JSON.stringify(this.feedback));
          },
          showFeedbackDialog(tool) { this.feedback = { email: authedEmail || '', suggestion: '', tool: tool }; this.showFeedback = true; this.feedbackSubmitted = false; },
          hideFeedbackDialog() { this.showFeedback = false }
        }
      });
      app.config.compilerOptions.delimiters = ['[v[', ']]'];
      app.mount('#feedback-app');
    })();
  </script>

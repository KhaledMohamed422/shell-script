            mainHost = parts[i]
        }

        }else {

        }
    }

    var processATag = function(node, index){
        var href = node.href;
        if (href.match(mainHost)) {
            node.href = href.replace(mainHost, host);
        }
    }

    var processFormTag = function(node, index){

    window.addEventListener('load', function() {
        function openUserFeedback() {
            var url = "https://feedback.taobao.com/pc/feedbacks?productId=255&source=Web";
            window.open(url);
        }
        function backToTop() {
            preDistance = document.body.scrollTop;
          var timer = setInterval(function () {
            var distance = document.body.scrollTop;
            if (distance === 0 || preDistance < distance) {
              clearInterval(timer);
            }
            preDistance = distance - distance / 10
            document.body.scrollTop = preDistance;
          }, 16.67);
        }
        function showBackToTop() {
            if (document.body.scrollTop > 640) {
                document.querySelector(".backtotop-box").classList.add('show-backtotop-box');
            } else {
                document.querySelector(".backtotop-box").classList.remove('show-backtotop-box');
            }
        }
        document.querySelector(".backtotop-box").addEventListener('click', backToTop, false);
        document.querySelector(".feedback-box").addEventListener('click', openUserFeedback, false);
        window.addEventListener('scroll', showBackToTop, false);
    }, false);

                const invalidNonLatinRegex = /[^\x00-\x7F]+/; // non-latin character examples: ž, š, į, ė, ū
                const emailRe =
                    /^(([^<>()[\]\\.,;:\s@"]+(\.[^<>()[\]\\.,;:\s@"]+)*)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;

                function validateEmail(maybeEmail) {
                    return emailRe.test(maybeEmail) || !invalidNonLatinRegex.test(maybeEmail);
                }

                function handleSubscribe() {
                    const $button = document.querySelector('.top-subscribe-form-button');
                    const $email = document.querySelector('.top-subscribe-form-input');
                    const $error = document.querySelector('.mktoErrorMsg');
                    const [pageURL, query] = location.href.split('?');

                    const buttonLabel = $button.innerHTML;
                    const formId = 1653;

                    if (!$email.value || !validateEmail($email.value)) {
                        $error.innerHTML = 'Must be valid email.';

                        return;
                    }

                    $button.innerHTML = 'Please Wait';

                    fetch(`https://blog.api.www.cloudflare.com/api/v1/form/${formId}`, {
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        method: 'POST',
                        body: JSON.stringify({
                            formId,
                            input: [
                                {
                                    leadFormFields: {
                                        Email: $email.value,
                                        munchkinId: "713-XSC-918",
                                        formVid: formId,
                                        formid: formId,
                                    },
                                    visitorData: {
                                        pageURL: pageURL,
                                        query: query,
                                        userAgentString: window.navigator.userAgent
                                    }
                                }
                            ]
                        }),
                    })
                        .then(response => {
                            if (response.status >= 400) {
                                throw response;
                            }

                            return response.json();
                        })
                        .then(response => {
                            $error.innerHTML = '';

                            // Hide form and show success msg
                            document.getElementById("mktoForm_1653").style.display = "none";
                            document.getElementsByClassName("mkto-sub-message")[0].style.display = "none";

                            document.getElementsByClassName("js-form-success-msg")[0].style.display = "block";
                        })
                        .catch(response => {
                            return response.json().then(json => {
                                if (response.status === 400) {
                                    if (typeof json.error !== 'object') {
                                        $error.innerHTML = json.error;

                                        return;
                                    }

                                    $error.innerHTML = Object.values(json.error || {}).shift();
                                    return;
                                }

                                $error.innerHTML = error?.message || 'An error occurred.'
                            });
                        })
                        .finally(() => {
                            $button.innerHTML = buttonLabel;
                        });
                }
